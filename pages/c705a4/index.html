# Git log格式化

# 参数

参数 作用
-p / --patch 按补丁形式显示diff信息
--oneline/--pretty=oneline 紧凑展示日志信息,让日志显示为一行,commtid未必是简化
--abbrev-commit commit_id按前7位展示
--graph 展示分支线,使merge一眼可见
--all 展示所有分支
--stat 展示commit对应修改的文件,以及修改的行数, -b/w 忽略空格
--relative-date 展示相比当前时间的相对时间间隔
--color 支持自定义颜色
--pretty=format 自定义日志格式
--name-only 显示每个提交更新的文件列表
--decorate 展示标签和分支的组成(似乎已没什么作用?)

git shortlog -s -n统计人员与行数

# 日志格式

参数 含义
%h 简短散列值
%p 展示父对象简短散列值
%an 作者名称author name
%cn 提交者名称
%cd 提交日期
%cr 相对于当前时间的提交日期
%s 修改内容或提交主题

# filter过滤日志

参数 作用
-2 / -n 2 显示2条日志
--skip=n 跳过前面的n条
--since, --after 指定日期后的提交 不含当天
--until , --before 指定日期前的提交
--author, --commiter 指定作者/提交者
--grep 变更记录字符串的提交,
-S 变更内容提交
--merges , --no-merges 合并提交,非合并提交
--max/max-parents=n max=0初始提交, max=1相当于--no-merges,min=2相当于--merges

filter还可配合展示log的相关参数,如-p / --patch同时打印变更内容 -i: 忽略大小写

# 附录-配合日志查看的自定义shell

  • 单行日志查看git-log-oneline
#!/bin/sh
##############################
# used for customize git log format
# Non zero exit codes:
#   1 - If not in a git repository
##############################

if [ ! "$(git rev-parse --git-dir 2> /dev/null)" ]; then
  echo Not in a git repository

  exit 1
fi

git log --pretty=oneline --abbrev-commit --color --pretty=format:"%Cred%h%Creset %C(yellow)%d%Creset %s %Cgreen(%cr) %C(bold blue)<%an>%Creset" --relative-date --decorate --graph "$@"

  • 查看当前分支与master/main主分支的差异git-log-from-main

(直接比较两个分支的差异,非merge-base比较)

#!/bin/sh

##############################
# used for show logs in this branch relative to main/master branch
# Non zero exit codes:
#   1 - If not in a git repository
#   2 - If neither git branch main nor master is found
##############################

if [ ! "$(git rev-parse --git-dir 2> /dev/null)" ]; then
  echo Not in a git repository

  exit 1
fi

if [ "$(git rev-parse -q --verify main)" ]; then
  git-log-oneline main^..HEAD "$@"

  exit 0
elif [ "$(git rev-parse -q --verify master)" ]; then
  git-log-oneline master^..HEAD "$@"

  exit 0
else
  echo 'Neither main nor master branch found'

  exit 2
fi
