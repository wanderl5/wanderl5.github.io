# Durations

# 概念

  • 描述时间差,内部实现上是一个模板类,可以表示几秒钟、几分钟或者几个小时的时间间隔
  • duration内部维护了周期个数rep和周期period(ratio)，两者结合用来表示间隔时间(注意ratio必须在编译阶段使用明确数字)
// duration原型
template <class Rep, class Period = ratio<1> > class duration;
1
2
  • ratio构造:
    • 第一个参数代表分子_Num，第二个代表分母_Den，两者表示一个通用的比率类型
    • 分母默认为1，因此ratio<60>代表60秒，ratio<1, 1000>代表0.001秒
// ratio原型
template<intmax_t _Num, intmax_t _Den = 1>
struct ratio;

# 应用

  • 线程sleep指定时间
#include <iostream>
#include <stdio.h>
#include <chrono>
#include <thread>
#include <sys/time.h>
using namespace std;

int main(int argc, char *argv[])
{
    using namespace std::chrono;
    for (int i = 0; i < 10; i++)
    {
        std::cout << "i: " << i << "..." << std::endl;
        steady_clock::time_point t1 = steady_clock::now();
        std::this_thread::sleep_for(chrono::minutes(1));       // 1m
        std::this_thread::sleep_for(chrono::seconds(5));       // 5s
        std::this_thread::sleep_for(chrono::milliseconds(30)); // 30ms
        steady_clock::time_point t2 = steady_clock::now();

        duration<double> time_span = duration_cast<duration<double>>(t2 - t1);
        // It took me 65.0375 seconds.
        std::cout << "It took me " << time_span.count() << " seconds." << std::endl;
    }
}

#include <stdio.h>  
#include <chrono>  
#include <thread>
#include <sys/time.h>
using namespace std;  

int main(int argc, char *argv[])  
{  
  struct timeval tv;
  gettimeofday(&tv, NULL);

  std::this_thread::sleep_for(chrono::minutes(1)); // 1m
  std::this_thread::sleep_for(chrono::seconds(5)); // 5s
  std::this_thread::sleep_for(chrono::milliseconds(30)); // 30ms

  struct timeval tv_now;
  gettimeofday(&tv_now, NULL);
  long interval = (tv_now.tv_sec - tv.tv_sec) * 1000 + (tv_now.tv_usec - tv.tv_usec)/1000; // ms
  printf("now: %ld\n", interval); // 65 * 1000 + 30 = 65030左右,实际是64991
}

# Time points/Clock

  • time_point模板类，表示一个时间点，如生日、今天日落时刻等，通过一个相对epoch的时间间隔duration来实现
  • Time points可通过time_point_cast转换为各类时间
  • timepoint对象只有一个成员duration,该值可以通过time_since_epoch()取得
#include <iostream>
#include <ratio>
#include <chrono>

int main ()
{
  using namespace std::chrono;
  typedef duration<int,std::ratio<60*60*24>> days_type;
  time_point<system_clock,days_type> today = time_point_cast<days_type>(system_clock::now());
  std::cout << today.time_since_epoch().count() << " days since epoch" << std::endl;

  return 0;
}

  • std::chrono::system_clock 它表示当前的系统时钟:

    • high_resolution_clock: 高精度时钟，实际上是system_clock或者steady_clock的别名,最小精度是纳秒

#include <iostream>
#include <ctime>
#include <chrono>
using namespace std;  

int main ()
{
    using std::chrono::system_clock;

    std::chrono::duration<int,std::ratio<60*60*24> > one_day (1);

    //current time(epoch精确到纳秒)
    system_clock::time_point today = system_clock::now();
    //tomorrow time (epoch)
    system_clock::time_point tomorrow = today + one_day;

    //convert time type   
    time_t tmTomorrow = system_clock::to_time_t ( tomorrow );    

    //time string,local format
    std::cout << "tomorrow will be(local time format): " << ctime(&tmTomorrow) << std::endl;

    // chrono epoch convert to readable time
    struct tm p;
    localtime_r(&tmTomorrow, &p); /*转换为struct tm结构的当地时间*/  
    printf("%04d-%02d-%02d %02d:%02d:%02d ", 1900 + p.tm_year, 1 + p.tm_mon, p.tm_mday,
        p.tm_hour, p.tm_min, p.tm_sec);

    return 0;
}

# steady_clock

  • steady_clock 专门用于计算时间差的工具，steady_clock 类只有一个静态函数now(),用于获取当前的时间
#include <iostream>
#include <chrono>

int main()
{
    using namespace std::chrono;
    typedef std::chrono::steady_clock STEADY_CLOCK;
    std::cout << "print 1000 stars" << std::endl;
    STEADY_CLOCK::time_point t1 = STEADY_CLOCK::now();    
    for (int i=1; i<=1000; ++i)
    {
        std::cout << "*";
        if (0 == i % 50) 
        {
            std::cout << "\n";
        }
    }
    std::cout << std::endl;
    auto t2 = STEADY_CLOCK::now();

    //毫秒
    auto dTimeSpan = duration<double,std::milli>(t2-t1);
    std::cout << "print star time span : " << dTimeSpan.count() << "ms\n";   
}

  • 实现IntervalTimer自动计时(析构时自动计算)
#include <iostream>
#include <chrono>
#include <string>
#include <thread>

class IntervalTimer
{
public:
    typedef enum {
        ttS = 's',  // 秒
        ttMS = 'm', // 毫秒
        ttUS = 'u'  // 微秒
    }TimeType;
private:
    std::chrono::steady_clock::time_point t1;
    std::chrono::steady_clock::time_point t2;
    std::string mark;
    TimeType timeType;
;

public:
    IntervalTimer(std::string mark, TimeType tt = ttMS) { 
      this->mark = mark;
      this->timeType = tt;
      t1 = std::chrono::steady_clock::now(); 
    }
    ~IntervalTimer()
    {
        t2 = std::chrono::steady_clock::now();
        if (timeType == ttS){
            auto dTimeSpan = std::chrono::duration<double, std::ratio<1>>(t2 - t1);
            std::cout << "[" << mark << "]time span : " << dTimeSpan.count() << " s\n";
        } else if (timeType == ttMS){
            auto dTimeSpan = std::chrono::duration<double, std::milli>(t2 - t1);
            std::cout << "[" << mark << "]time span : " << dTimeSpan.count() << " ms\n";
        }else if (timeType == ttUS){
            auto dTimeSpan = std::chrono::duration<double, std::micro>(t2 - t1);
            std::cout << "[" << mark << "]time span : " << dTimeSpan.count() << " us\n";
        }

    }
};

int main()
{
    IntervalTimer t("画星星", IntervalTimer::ttUS);
    for (int i = 1; i <= 1000; ++i)
    {
        std::cout << "*";
        if (0 == i % 50)
        {
            std::cout << "\n";
        }
        auto interval = std::chrono::milliseconds(5);
        std::this_thread::sleep_for(interval);
    }
    std::cout << std::endl;

    return 0;
}

# 以计时器停滞线程

  • uration和Timepoint可用于线程或程序（即主线程）的停滞（block）。停滞可以是无条件的，也可以指定最大时间段，或等待一个lock或某条件成立，或等待另一线程结束
  • 提供的操作如下：
    • sleep_for()和sleep_until()：由this_thead提供用以停滞线程
    • try_lock_for()和try_lock_until()：用来在等待一个mutex时指定最大时间段
    • wait_for()和wait_until()：用来在等待某条件成立或等待一个future时指定最大时间段
  • 所有以...for()结尾的函数都会用到一个duration，所有以...until()结束的函数都会用到一个timepoint。例如：
    • this_thread::sleep_for(chrono::seconds(10)); //会停滞当前线程（不无可能是主线程）10秒钟
    • this_thread::sleep_until(chrono::system_clock::now()+chrono::seconds(10));
