# yum install zsh

# 将github内容调整为gitee, 详见以下内容
# Default settings
REPO=${REPO:-ohmyzsh}
REMOTE=${REMOTE:-https://gitee.com/mirrors/${REPO}.git}

cd oh-my-zsh/tools
./install.sh

# 插件和主题安装

# 插件:命令行提示(zsh-autosuggestions)

  • 插件用途说明:
    • 每次输入命令后,会有代码提示,如果是自己想要的直接按->键即可选择此命令
    • 如果命令不是你想要的,可以tab继续查找;
cd ~/.oh-my-zsh/plugins
git clone https://gitee.com/phpxxo/zsh-autosuggestions.git ~/.oh-my-zsh/plugins/zsh-autosuggestions
1
2

# 插件:高亮(zsh-syntax-highlighting)

  • 已下载文件安装
cd ~/.oh-my-zsh/plugins
# 请按实际路径调整
cp ~/Documents/soft/zsh/zsh-syntax-highlighting.zip .
unzip zsh-syntax-highlighting.zip
mv zsh-syntax-highlighting-master zsh-syntax-highlighting
1
2
3
4
5
  • 从github下载代码安装 https://github.com/zsh-users/zsh-syntax-highlighting

注意将文件夹名称调整为跟插件名称一致，删除-master

# 插件:autojump

# mac

  • 下载
# 建议直接从gitee或github下载文件到/opt/autojump下;(如网络问题导致无法下载,可手动下载后放置到指定路径下)
# git clone https://github.com/joelthelion/autojump.git (zip包已存放在soft/zsh下)
cd ~/.oh-my-zsh/plugins
cp /doc/soft/zsh/autojump.zip .
unzip autojump.zip
cd autojump
chmod 755 ./install.py
chmod 755 ./bin/autojump

# 安装主题:powerlevel10k

  • 安装
git clone --depth=1 https://gitee.com/romkatv/powerlevel10k.git ${ZSH_CUSTOM:-$HOME/.oh-my-zsh/custom}/themes/powerlevel10k
1
  • 配置.zshrc启用theme
ZSH_THEME="powerlevel10k/powerlevel10k"
1
  • 自定义配置 (ubuntu不需要,zsh进入直接可以进入配置)
# mac下配置好zshrc后，首次登录zsh,会提示配置(注: 字体可能会安装不了)
# 再次配置可以
p10k configure

# 配置修改: .zshrc配置(插件和其他配置项)

  • 找到对应配置修改
#设置主题
# ZSH_THEME="sonicradish"

#禁用自动更新(默认是自动更新,去除#号注释,即可关闭自动更新)
zstyle ':omz:update' mode disabled  # disable automatic updates

#设置加载插件
plugins=(
  git
  zsh-autosuggestions
  zsh-syntax-highlighting
  autojump
)
