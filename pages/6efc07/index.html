WSL-GCC-Debug

# Config Deubg in WSL

# 概述

本章内容主要实现WSL下C，CPP文件的编译和调试，同时包括使用CMAKE是实现统一编译

# 配置vscode wsl

  • 安装插件Remote - WSL
  • 在WSL Remote侧边栏下新增WSL服务器，连接成功后即可如打开本地文件一样使用了
  • 安装插件C/C++插件，支持inteligence
  • 安装CMake插件(Microsoft版本)

# GCC调试C语言

# 设置代码查看Inteligence - c_cpp_properties.json

键入Ctrl + Shift + P打开命令面板。输入“C/C++ edit configurations”，选择UI的直观简单些。 （以下以RDFT为例） 在compiler下选择/usr/bin/gcc或g++,视C/C++有差异,配置后查看配置"C/C++ edit configurations"选择JSON

{
    "configurations": [
        {
            "name": "Linux",
            "includePath": [
                "${workspaceFolder}/**"
            ],
            "defines": [],
            "compilerPath": "/usr/bin/gcc",
            "cStandard": "c11",
            "cppStandard": "c++14",
            "intelliSenseMode": "gcc-x64"
        }
    ],
    "version": 4
}

# 配置编译任务-task.json

打开命令面板。输入task，选择任务：配置默认生成任务，使用模板创建tasks.json文件、选择Others，形成task.json,根据自己的文件重新修改

{
    "version": "2.0.0",
    "windows": {
        "options": {
            "shell": {
                "executable": "bash.exe",
                "args": [
                    "-c"
                ]
            }
        }
    },
    "tasks": [        
        {
            "type": "shell",
            "label": "C/C++: gcc build active file",
            "command": "/usr/bin/gcc",
            "args": [
                "-g",
                "${fileDirname}/main.c",
                "${fileDirname}/extract.c",
                "${fileDirname}/myFunction.c",
                "${fileDirname}/cJSON.c",
                "-o",
                "/root/dft3.1/bin/linux/RDFT",
                "-D_DEBUG",
                "-D_UNIX",
                "-DUNIX",
                "-D_PTHREAD_SAFE",
                "-D_ORACLE",
                "-D_LINUX",
                "-I/opt/oracle/precomp/public",
                "-L/opt/oracle/lib",
                "-lclntsh",
                "-lm"
            ],
            "options": {
                "cwd": "${workspaceFolder}"
            },
            "problemMatcher": [
                "$gcc"
            ],
            "group": {
                "kind": "build",
                "isDefault": true
            },
            "detail": "Task generated by Debugger."
        }
    ]
}

# 调试-launch.json

写一个C++文件，设置好断点，按F5就可以开始调试了。 设置好调试时的args，按下F5会有个提示，选择C++(GDB/LLDB) 选择gcc build and debug active file，配置文件会相应调整;

{
    // Use IntelliSense to learn about possible attributes.
    // Hover to view descriptions of existing attributes.
    // For more information, visit: https://go.microsoft.com/fwlink/?linkid=830387
    "version": "0.2.0",
    "configurations": [
        {
            "name": "gcc - Build and debug active file",
            "type": "cppdbg",
            "request": "launch",
            "program": "/root/dft3.1/bin/linux/RDFT",
            "args": [
                "-F/root/dft3.1/bin/linux/RDFT.json"
            ],
            "stopAtEntry": false,
            "cwd": "${workspaceFolder}",
            "environment": [],
            "externalConsole": false,
            "MIMode": "gdb",
            "setupCommands": [
                {
                    "description": "Enable pretty-printing for gdb",
                    "text": "-enable-pretty-printing",
                    "ignoreFailures": true
                }
            ],
            "preLaunchTask": "C/C++: gcc build active file",
            "miDebuggerPath": "/usr/bin/gdb"
        }
    ]
}

# CMAKE实现统一调试

# CMakeLists.txt

  • 编写CMakeLists.txt,实例见下
  • 在wsl下mkdir build,cd到build下cmake ../, 然后make通过 （此步骤不作，目前vscode无法编译和调试，也许跟设置有关，暂放一边）
  • 设置CMakeKit，可以CMake: Scan for kits选择GCC4.8.5
  • 其他调试可以同普通编译调试
CMAKE_MINIMUM_REQUIRED (VERSION 3.1) #最低要求的CMake版本
PROJECT(basic) # 项目名称

# PROJECT_SOURCE_DIR - where the main CMakeLists.txt locate
#INCLUDE_DIRECTORIES(${PROJECT_SOURCE_DIR}/include/common)

# 设置为Debug属性
SET (CMAKE_BUILD_TYPE  "Debug")
# 设置输出路径,build目录下bin
SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)

# 设置包含路径
#INCLUDE_DIRECTORIES($ENV{PUB_SMDEV}/include)

# 设置预处理头
#ADD_DEFINITIONS(-g -fPIC -std=c++11 -D_DEBUG -D_UNIX -D_PTHREAD_SAFE -D_ORACLE -D_LINUX)
ADD_DEFINITIONS(-g -std=c++11)
#链接lib
#FIND_LIBRARY(SM_LIB libsm.a $ENV{PUB_SMDEV}/lib64/linux)

########### 指定编译源码
#当前路径下所有cpp纳入编译
#AUX_SOURCE_DIRECTORY(. DIR_SRCS)

#建立变量SRC_FILE为目录下指定文件列表
FILE(GLOB SRC_FILE ${PROJECT_SOURCE_DIR}/src/main.cpp) 

########### 编译可执行文件
#ADD_EXECUTABLE(${PROJECT_NAME} ${DIR_SRCS})
ADD_EXECUTABLE (${PROJECT_NAME} ${SRC_FILE}) # 要求编译可执行文件
# 链接lib
#TARGET_LINK_LIBRARIES(${PROJECT_NAME} Mylib ${SM_LIB} ${LANG_LIB} ${HANCE_LIB} ${XNET_LIB})

# task.json

{
    "tasks": [
        {
            "label": "mkdir",
            "type": "shell",
            "command": "mkdir",
            "args": [
                "-p",
                "build"
            ],
            "options": {
                "cwd": "${workspaceFolder}"
            }
        },
        {
            "label": "cmake",
            "type": "shell",
            "command": "cmake",
            "args": [
                ".."
            ],
            "options": {
                "cwd": "${workspaceFolder}/build"
            },
            "dependsOn": [
                "mkdir"
            ]
        },
        {
            "label": "make", //需要和 lauch.json 中的  "preLaunchTask" 相同
            "type": "shell",
            "command": "make",
            "args": [],
            "options": {
                "cwd": "${workspaceFolder}/build"
            },
            "dependsOn": [
                "cmake"
            ] //执行当前命令前先执行 label 为 cmake 的命令
        }
    ],
    "version": "2.0.0"
}
