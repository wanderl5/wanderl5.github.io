# 8 结构体

# 8.1 结构体基本概念

结构体属于用户==自定义的数据类型==，允许用户存储不同的数据类型

# 8.2 结构体定义和使用

语法：struct 结构体名 { 结构体成员列表 }； 通过结构体创建变量的方式有三种：

  • struct 结构体名 变量名
  • struct 结构体名 变量名 = { 成员1值 ， 成员2值...}
  • 定义结构体时顺便创建变量
class cstudent{
int age;
string name;
};
struct student{
int age;
string name;
};

int main() {
    student st;
    st.age = 20;

    cstudent cst;
    // cst.age = 20; // inaccessible
    
	return 0;
}

# 8.5 结构体嵌套结构体

作用： 结构体中的成员可以是另一个结构体 **例如：**每个老师辅导一个学员，一个老师的结构体中，记录一个学生的结构体

//学生结构体定义
struct student
{
	string name;  //姓名
	int age;      //年龄
	int score;    //分数
};

//教师结构体定义
struct teacher
{
	int id; //职工编号
	string name;  //教师姓名
	int age;   //教师年龄
	struct student stu; //子结构体 学生
};
int main() {
	struct teacher t1;
	t1.id = 10000;
	t1.name = "老王";
	t1.age = 40;

	t1.stu.name = "张三";
	t1.stu.age = 18;
	t1.stu.score = 100;

	cout << "教师 职工编号： " << t1.id << " 姓名： " << t1.name << " 年龄： " << t1.age << endl;
	cout << "辅导学员 姓名： " << t1.stu.name 
    	<< " 年龄：" << t1.stu.age 
    	<< " 考试分数： " << t1.stu.score << endl;

	return 0;
}

# 8.6 结构体做函数参数

**作用：**将结构体作为参数向函数中传递

传递方式有两种：

  • 值传递
  • 地址传递

struct student
{
	string name;  
	int age;     
	int score;    
    std::string toString() const {
        // name = "nihao"; // change member value in const function is not allowed;
        lang::mstring mstr;
        mstr << "name:" << name << " age:" << age << " score:" << score;
        cout << mstr.str() << endl;
        return mstr.str();
    }
};

void fun1(student stu )
{
	stu.age = 28;
}
void fun2(student *stu)
{
	stu->age = 28;
}

int main() {
    CY_FUN::printSep("origin");
	student stu = { "zs", 18, 100};
    stu.toString();

    CY_FUN::printSep("pass by value");
	fun1(stu);
    stu.toString();

	CY_FUN::printSep("pass by address");
	fun2(&stu);
    stu.toString();

	return 0;
}

# 8.7 结构体中 const使用场景

**作用：**用const来防止误操作

//学生结构体定义
struct student
{
	string name;  //姓名
	int age;      //年龄
	int score;    //分数
};

//const使用场景
void printStudent(const student *stu) //加const防止函数体中的误操作
{
	//stu->age = 100; //操作失败，因为加了const修饰
	cout << "姓名：" << stu->name << " 年龄：" << stu->age << " 分数：" << stu->score << endl;
}

int main() {
	student stu = { "张三",18,100 };
	printStudent(&stu);
	system("pause");

	return 0;
}

# 8.8.1 案例1

案例描述：

学校正在做毕设项目，每名老师带领5个学生，总共有3名老师，需求如下

设计学生和老师的结构体，其中在老师的结构体中，有老师姓名和一个存放5名学生的数组作为成员

学生的成员有姓名、考试分数，创建数组存放3名老师，通过函数给每个老师及所带的学生赋值

最终打印出老师数据以及老师所带的学生数据。

示例：

#include <ctime>
struct Student
{
	string name;
	int score;
};
struct Teacher
{
	string name;
	Student sArray[5];
    std::string toString() const {
        // change member value in const function is not allowed;
        lang::mstring mstr;
        mstr << "name:" << name << "\n";
        for (int i = 0; i < 5; i++){
            mstr << "\t" << sArray[i].name << " " << sArray[i].score << "\n";
        }
        // cout << mstr.str() << endl;
        return mstr.str();
    }
};

void Init(Teacher tArray[] , int len)
{
	string tName = "teacher";
	string sName = "stu";
	for (int i = 0; i < len; i++)
	{
		tArray[i].name = lang::TCharSet::Format("tea%d", i);		
		for (int j = 0; j < 5; j++)
		{
			tArray[i].sArray[j].name = lang::TCharSet::Format("stu%d", j);
			tArray[i].sArray[j].score = rand() % 41 + 60;
		}
	}
}

int main() {
	srand((unsigned int)time(NULL)); // random must use time as seed, or won't random any more
	Teacher tArray[3];
	int len = sizeof(tArray) / sizeof(tArray[0]);
	Init(tArray, len);
	CY_FUN::toStringCustom(tArray, len); 
	return 0;
}

# 8.8.2 案例2

案例描述：

设计一个英雄的结构体，包括成员姓名，年龄，性别;创建结构体数组，数组中存放5名英雄。

通过冒泡排序的算法，将数组中的英雄按照年龄进行升序排序，最终打印排序后的结果。

五名英雄信息如下：

		{"liubei",23,"male"},

		{"zhaoyun",21,"male"},

struct hero
{
	string name;
	int age;
	string sex;
    std::string toString() const {
        lang::mstring mstr;
        mstr << "name:" << name << " age:" << age << " sex:" << sex;
        return mstr.str();
    }
};
void bubbleSort(hero arr[] , int len)
{
	for (int i = 0; i < len - 1; i++)
	{
		for (int j = 0; j < len - 1 - i; j++)
		{
			if (arr[j].age > arr[j + 1].age)
			{
				hero temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;
			}
		}
	}
}
int main() {
	struct hero arr[5] =
	{
	    {"liubei",23,"male"},
		{"guanyu",22,"male"},
		{"zhangfei",20,"male"},
		{"zhaoyun",21,"male"},
		{"diaochan",19,"female"}
	};
	int len = sizeof(arr) / sizeof(arr[0]); //获取数组元素个数
	bubbleSort(arr, len);
	CY_FUN::toStringCustom(arr, len);

	return 0;
}
