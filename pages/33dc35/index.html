# VIMRC文件

~/.vimrc用于控制vim的参数设置,快捷键配置等;

# vimrc文件示例

"设置兼容性
set nocompatible
"tab长度
set ts=4

"记录修改日志
"""""""""""""""""""""""""""""""""""
function Log()
    "let strStart=printf("// modify by cy %s", $V_DateTime)
    let v_datetime = strftime("%Y%m%d")
    let strStart = printf("// modify by %s %s", $coder, v_datetime)
    let strEnd   = printf("// end modify by %s %s", $coder, v_datetime)
    "let strNote = "// Reason: "
    "call setline(line("."), strStart)
    call append(line("."), strStart)
    "call append(line(".") + 1, strNote)
    call append(line(".") + 1, strEnd)
endfunction

"nnoremap <silent> ;l :call Log()<CR><Enter>
nmap <silent> ;; :call Log() <CR>
"""""""""""""""""""""""""""""""""""

""记录日期
"""""""""""""""""""""""""""""""""""
function AddDate()
    let line = getline(".")
    let v_datetime = strftime("%Y/%m/%d")
    let repline = printf("%s %s", line, v_datetime)
    call setline(".", repline)
endfunction

"nnoremap <silent> ;l :call Log()<CR><Enter>
nmap <silent> <F7> :call AddDate() <CR>
"""""""""""""""""""""""""""""""""""


"Quick Comment and UnComment
nnoremap <silent> <F5> I//<SPACE><ESC>j
nnoremap <silent> <F6> 0f/dwj

nmap ;d :set nocindent <CR>:set noai<CR>
nmap ;e :set cindent <CR>:set ai<CR>

" -------- Status Bar
set showcmd

set laststatus=2
function! CurDir()
     let curdir = substitute(getcwd(), '/Users/amir/', "~/", "g")
     return curdir
endfunction

set statusline=\ %F%m%r%h\ %w\ \ CWD:\ %r%{CurDir()}%h\ \ \ Line,Col:\ %l/%L,%c
"----------- end Status Bar

"Show UTF
nmap ;u :set enc=UTF-8 <CR>
nmap ;a :set enc=GB2312 <CR>

"forbid auto comment
set formatoptions=tcmM
