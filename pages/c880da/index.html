# 全局组件

全局组件只要定义了处处可使用,会占用一定内存,性能相对差 组件可复用,但组件间独立,数据不共享

<script>
    const app = Vue.createApp({
        template:`
            <!-- 每个组件都是独立的 -->
            <counter/>
            <counter/>
            <counter/>
        `
    })
    app.component('counter', {
        data(){
            return {
                count : 1
            }
        },
        template: `
            <!-- 单行表达式 -->
            <div @click="count += 1">{{ count }} </div>
        `
    }),
    app.mount('#root');
</script>

使用局部组件,动态形成表格

<!-- head -->
<script src="https://unpkg.com/vue@3/dist/vue.global.js"></script>
<script>
    const app = Vue.createApp({
        data() {
            return {
                thead:['name', 'age'],
                tdata:[
                    {name:'zs', age:18},
                    {name:'ls', age:20},
                    {name:'ww', age:100},
                ],
            }
        },
        template:`
            <!-- 父子组件参数传递 -->
            <mytable :thead="thead" :tdata="tdata"></mytable>
        `
    })
    app.component('mytable', {
        <!-- 父子组件参数传递 -->
        props:['thead', 'tdata'],
        template: `
        <table border="1" bgcolor="#00FF00">
            <tr>
                <th v-for="item of thead" :key="item.id">{{item}}</th>
            </tr>
            <tr v-for="item_data of tdata" :key="item_data.id">
                <td>{{item_data.name}}</td>
                <td>{{item_data.age}}</td>
            </tr>
        </table>
        `
    }),
    app.mount('#root');
</script>

# 局部组件

局部组件仅在注册后才会引入,建议驼峰首字母大写命名

<script>
    const Counter={
        data(){
            return {
                count : 1
            }
        },
        template: `
            <div @click="count += 1">Counter:{{ count }} </div>
        `
    }
    const CounterAnother={
        data(){
            return {
                count : 2
            }
        },
        template: `
            <div @click="count += 1">CounterAnother:{{ count }} </div>
        `
    }
    const app = Vue.createApp({
        // 使用到的局部组件需显式声明,支持再次自定义命名
        // components: {
        //     'dell': Counter,
        //     'dell-2': CounterAnother,
        // },
        // template:`
        //     <dell/>
        //     <dell-2/>
        // `

        // 组件默认驼峰首字母大写,网页使用时横杠分割;不自定义命名时,使用默认名称自动映射
        components: {
            Counter,CounterAnother,
        },
        template:`
            <Counter/>
            <Counter-Another/>
        `
    });
    app.mount('#root');
</script>

# 简单消息传递(含函数)

  • 简单字符串传递
  • 变量绑定传递(v-bind 或 😃
  • 类型校验(Number, Boolean)
  • 函数参数传递
<script>
const Comp1={
    props:{
        var1: String,
        var2: Number,
        var3: Boolean,
        var4: Array,
        var5: Object,
        var6: Function,
    },
    methods:{
        handleClick(){
            this.var6();
        },
    },
    template: `
        <div @click="this.handleClick">Comp1:{{ var1 }}, {{ var2 }}, {{var3}}, {{var4}}, {{var5}}, {{var6}} </div>
    `
};
const app = Vue.createApp({
    data(){

            var2: 456,
            var3: true,

            var6: () => {alert("function is called.")},
        }
    },
    components: {
        Comp1,
    },
    template:`
        <Comp1 var1="123" :var2="var2" :var3="var3" :var4="var4" :var5="var5" :var6="var6"/>
    `
});
app.mount('#root');
</script>

# 数值校验

  • required, type, default, validator
<script>
const Comp1={
    props:{
        content: {
            // 不传值时会提示
            required: true,
            // 类型检查,不满足则会在console中提示
            type: Number,
            // 自定义校验,返回false则会在console中提示
            validator: function(value){
                return value < 1000;
            },
            // 未设置content的值则会使用此默认值
            default: function(){
                return 123;
            }
        }
    },
    template: `
        <div>Comp1:{{content}} </div>
    `
};
const app = Vue.createApp({
    data(){
        return{
            // content: 1234,
        }
    },
    components: {
        Comp1,
    },
    template:`
        <Comp1 :content="content" />
    `
});
app.mount('#root');
</script>

# v-bind传值简化

<script>
const Comp1={
    props:{
        var1: String,
        var2: Number,
        var3: Boolean,
        var4: Array,
        var5: Object,
        var6: Function,
    },
    methods:{
        handleClick(){
            this.var6();
        },
    },
    template: `
        <div @click="this.handleClick">Comp1:{{ var1 }}, {{ var2 }}, {{var3}}, {{var4}}, {{var5}}, {{var6}} </div>
    `
};
const app = Vue.createApp({
    data(){
        return{
            // 参数整合,方便传参
            params:{
            var2: 456,
            var3: true,
            var4: ['123', '444'],
            var5: {name:'zs', age: 45},
            var6: () => {alert("function is called.")},
            }
        }
    },
    components: {
        Comp1,
    },

    template:`
        <!-- v-bind简化传参 -->
        <Comp1 v-bind="params"/>
    `
});
app.mount('#root');
</script>

# 单向数据流

# 子组件可以使用父组件的数据,但不能修改(主要是为了减少组件间的耦合)

<!-- 运行仅有readonly属性提示，数值不会变化 -->
<script>
const Comp1={
    props:['count'],
    // 无法修改,console提示属性只读
    template: `<div @click="count += 1">{{ count }} </div> `
};
const app = Vue.createApp({
    data(){
        return{  count: 1,  }
    },
    components: {   Comp1,   },
    // 传值时使用content-abc,接用驼峰命名
    template:` <Comp1 :count="count"/> `
});
app.mount('#root');
</script>

# 使用emit修改父组件属性

<script>
const Comp1={
    props:['count'],
    emits:{}
    methods:{
        addOne(){
            console.log("Comp1 addOne...");
            this.$emit('addOne'); // 注意必须驼峰
        },
    },
    template: `<div @click="addOne">{{ count }} </div> `
};
const app = Vue.createApp({
    data(){    return{  count: 1,  }
    },
    methods:{
        handleAddOne(){
            console.log("handleAddOne in app");
            this.count += 1;
        },
    },
    components: {   Comp1,   },
    // 接收子组件的事件,需要用横杠来接收
    template:` <Comp1 :count="count" @add-one="handleAddOne"/> `
});
app.mount('#root');
 </script>

# 带参数emit

<script>
const Comp1={
    props:['count'],
    emits:{
        addOne: (count) => { // 此处接收组件传递的参数
            if (count > 5) return false;
            return true;
        }
    },
    methods:{
        addOne(){
            console.log("Comp1 addOne...");
            this.$emit('addOne', this.count); // 带参数的emit
        },
    },
    template: `<div @click="addOne">{{ count }} </div> `
};
const app = Vue.createApp({
    data(){    return{  count: 1,  }
    },
    methods:{
        handleAddOne(){
            console.log("handleAddOne in app");
            this.count += 1;
        },
    },
    components: {   Comp1,   },
    // 接收子组件的事件,需要用横杠来接收
    template:` <Comp1 :count="count" @add-one="handleAddOne"/> `
});
app.mount('#root');
</script>

# 变量绑定子修改父

<script>
const app = Vue.createApp({
    data(){
        return{  count: 1,  }
    },
    // v-model绑定变量
    template:` <Comp1 v-model="count"/> `  // 此用法必须使用modelValue,大小写敏感
    // template:` <Comp1  v-model:app="count"/> ` // 也可以使用自定义值,在子组件中都使用自定义值替换modelValue即可
});
app.component('Comp1',{
    props:['modelValue'],
    methods:{
        handleClick(){
            this.$emit('update:modelValue', this.modelValue * 3); // 此处在子组件更新父组件的值,写具体如何更新
        }
    },
    template: `<div @click="handleClick"> {{ modelValue }} </div> `
});
app.mount('#root');
</script>

# (多值绑定)变量绑定子修改父

<script>
const app = Vue.createApp({
    data(){
        return{
            count1: 1,
            count2: 2,
        }
    },
    // v-model绑定变量
    template:` <Comp1 v-model:count1="count1" v-model:count2="count2" /> `
});
app.component('Comp1',{
    props:['count1', 'count2'],
    methods:{
        handleClick1(){
            this.$emit('update:count1', this.count1 + 3); // 此处在子组件更新父组件的值,写具体如何更新
        },
        handleClick2(){
            this.$emit('update:count2', this.count2 * 3); // 此处在子组件更新父组件的值,写具体如何更新
        },
    },
    template: `
        <div @click="handleClick1"> {{ count1 }} </div>
        <div @click="handleClick2"> {{ count2 }} </div>
    `
});
app.mount('#root');
</script>

# 自定义v-model修饰符

  • 使用默认modelValue
<script>
const app = Vue.createApp({
    data(){
        return{
            msg: "zs",
        }
    },
    // v-model绑定变量
    template:` <Comp1 v-model.uppercase="msg" /> `
});
app.component('Comp1',{
    props:{
        'modelValue': String,
        'modelModifiers': { // 默认不作任何调整
            default: () => ({})
        }
    },
    methods:{
        handleClick1(){
            let newValue= this.modelValue;
            console.log(newValue, this.modelModifiers);
            if (this.modelModifiers.uppercase){
                console.log('upppercase');
                newValue = newValue.toUpperCase();
            }
            this.$emit('update:modelValue', newValue);
        },
    },
    template: `
        <div @click="handleClick1"> {{ modelValue }} </div>
    `
});
app.mount('#root');
</script>

const app = Vue.createApp({

        <div @click="handleClick1"> {{ msg }} </div>
    `
});
