# 主程序逻辑

import os
import zipfile
import ftplib
import datetime
import time
import argparse
import logging
import sys

from glob import glob
from datetime import datetime, timedelta

class FTPDebugWrapper:
    """FTP调试包装器，记录所有FTP命令和响应"""
    def __init__(self, ftp, logger):
        self.ftp = ftp
        self.logger = logger

    def __getattr__(self, name):
        def wrapper(*args, **kwargs):
            # 记录发送的命令
            self.logger.debug(f"FTP 命令: {name} {args} {kwargs}")

            # 获取原始方法
            method = getattr(self.ftp, name)

            # 执行方法并捕获响应
            try:
                result = method(*args, **kwargs)
                self.logger.debug(f"FTP 响应: {result}")
                return result
            except ftplib.all_errors as e:
                self.logger.error(f"FTP 错误: {str(e)}")
                raise

        return wrapper

def setup_logging(log_file):
    """配置日志记录"""
    logging.basicConfig(
        level=logging.DEBUG,  # 设置为DEBUG以获取FTP调试信息
        format='%(asctime)s - %(levelname)s - %(message)s',
        handlers=[
            logging.FileHandler(log_file),
            logging.StreamHandler()
        ]
    )
    return logging.getLogger('log_backup')

def parse_arguments():
    """解析命令行参数"""
    parser = argparse.ArgumentParser(description='日志打包和FTP传输工具')
    parser.add_argument('--source', required=True, help='要打包的日志文件夹路径')
    parser.add_argument('--ftp_host', required=True, help='FTP服务器地址')
    parser.add_argument('--ftp_user', required=True, help='FTP用户名')
    parser.add_argument('--ftp_pass', required=True, help='FTP密码')
    parser.add_argument('--ftp_dir', help='FTP目标目录，默认为根目录', default='')
    parser.add_argument('--keep_days', help='备份多少天内的日志文件', default='1:0')
    parser.add_argument('--zip_prefix', help='ZIP文件前缀', default='qe')
    parser.add_argument('--log_file', help='日志文件名称', default='log_backup')
    parser.add_argument('--log_path', help='日志文件路径', default='log')
    parser.add_argument('--extensions', type=str, required=True,
                      help='逗号分隔的后缀名列表，如 ".log,.gz,.txt"')
    # parser.add_argument('--max_log_size', type=int,
    #                    help='本地日志文件最大大小(MB)，超过则轮转-暂无用b
    parser.add_argument('--debug_ftp', action='store_true',
                       help='启用详细的FTP调试信息')
    return parser.parse_args()

def clear_log_file(file_path):
    """
    安全清空日志文件
    参数:
        file_path (str): 要清空的日志文件路径
    返回:
        bool: 操作是否成功
    """
    try:
        # 先检查文件是否存在
        if not os.path.exists(file_path):
            logger.warning(f'日志文件不存在: {file_path}')
            return False

        # 获取文件当前大小用于记录
        original_size = os.path.getsize(file_path)

        # 使用'r+'模式打开并清空
        with open(file_path, 'r+') as f:
            f.truncate(0)  # 将文件截断为0字节

        logger.info(f'成功清空日志文件: {file_path} (原大小: {original_size} bytes)')
        return True

    except PermissionError:
        logger.error(f'权限不足，无法清空文件: {file_path}')
        return False
    except Exception as e:
        logger.error(f'清空日志文件失败: {file_path}, 错误: {str(e)}', exc_info=True)
        return False

def rotate_logs(log_file, max_size_mb):
    """日志文件轮转"""
    try:
        if os.path.exists(log_file):
            size_mb = os.path.getsize(log_file) / (1024 * 1024)
            if size_mb > max_size_mb:
                timestamp = datetime.datetime.now().strftime('%Y%m%d_%H%M%S')
                backup_name = f"{log_file}.{timestamp}.bak"
                # os.rename(log_file, backup_name)
                clear_log_file(log_file)
                logger.info(f'日志文件轮转: {log_file} -> {backup_name}')
    except Exception as e:
        logger.error(f'日志轮转失败: {str(e)}')


"""
    校验文件后缀
    exts: log, log, gz
"""
def valid_ext(file_path, exts):
    file_ext = os.path.splitext(file_path)[1]
    # if file.lower().endswith(('.log', '.LOG')):
    allowed_extensions = [ext.strip() for ext in exts.split(',')]
    # print(f"允许的后缀名: {allowed_extensions}")
    if file_ext in allowed_extensions:
        return True
    return False

"""
keep_days: 备份天数, 格式 1:0 (开始备份日期:结束备份日期,相对于当日往前偏移)
"""
def get_beg_end_dt(keep_days):
    beg_str, end_str = keep_days.split(':')  # 分割字符串
    beg = int(beg_str)
    end = int(end_str)
    # 获取当前时间
    now = datetime.now()
    """获取目录中昨天被修改的文件列表"""
    # 起始日期零点
    dt_beg = now - timedelta(days=beg)
    dt_beg_real = datetime(dt_beg.year, dt_beg.month, dt_beg.day)

    # 结束日期23:59:59
    dt_end = now - timedelta(days=end)
    dt_end_real = datetime(dt_end.year, dt_end.month, dt_end.day,  23, 59, 59, 999999)
    print(f"开始时间: {dt_beg_real}, 结束时间: {dt_end_real}")

    return dt_beg_real,dt_end_real

def create_zip(zip_filename, source_dir, keep_days, extensions):
    """创建ZIP压缩包，只包含最近keep_days天的文件"""
    dt_beg_real,dt_end_real = get_beg_end_dt(keep_days)
    # cutoff_time = time.time() - (keep_days * 86400)
    file_count = 0
    try:
        with zipfile.ZipFile(zip_filename, 'w', zipfile.ZIP_DEFLATED) as zipf:
            for root, dirs, files in os.walk(source_dir):
                for file_path in files:
                    file_path_full = os.path.join(root, file_path)
                    # 只处理日志文件（按扩展名过滤）
                    if not valid_ext(file_path, extensions):
                        continue
                    file_time = datetime.fromtimestamp(os.path.getmtime(file_path_full))
                    # logger.debug(f'{file_path}, {file_time} ...')
                    if dt_beg_real <= file_time and file_time <= dt_end_real:
                        arcname = os.path.relpath(file_path_full, start=source_dir) # 在ZIP中保持相对路径
                        zipf.write(file_path_full, arcname)
                        file_count += 1
                        logger.debug(f'已添加: {arcname}, {file_time}')

            logger.info(f'成功创建ZIP文件: {zip_filename}，包含 {file_count} 个文件')
            return True
    except Exception as e:
        logger.error(f'创建ZIP文件失败: {str(e)}')
        return False

def upload_via_ftp(zip_filename, ftp_host, ftp_user, ftp_pass, ftp_dir, debug_ftp=False):
    """通过FTP上传文件，带有详细的调试信息"""
    try:
        ftp = ftplib.FTP(ftp_host)

        # 启用调试输出
        if debug_ftp:
            ftp.set_debuglevel(2)  # 设置FTP调试级别为2（最大详细程度）
            ftp = FTPDebugWrapper(ftp, logger)  # 使用我们的调试包装器

        # 登录
        logger.info(f'正在连接到FTP服务器: {ftp_host}')
        ftp.login(ftp_user, ftp_pass)
        logger.info(f'已登录FTP服务器，欢迎消息: {ftp.getwelcome()}')

        # 切换到目标目录（如果指定）
        if ftp_dir:
            try:
                ftp.cwd(ftp_dir)
                logger.info(f'已切换到目录: {ftp_dir}')
            except ftplib.error_perm:
                # 目录不存在，尝试创建
                parts = ftp_dir.split('/')
                path = ''
                for part in parts:
                    if not part:
                        continue
                    path += '/' + part
                    try:
                        ftp.cwd(path)
                    except ftplib.error_perm:
                        logger.debug(f'尝试创建目录: {path}')
                        ftp.mkd(path)
                        ftp.cwd(path)
                logger.info(f'已创建并切换到目录: {ftp_dir}')

        # 上传文件
        logger.info(f'开始上传文件: {zip_filename}')
        with open(zip_filename, 'rb') as f:
            ftp.storbinary(f'STOR {os.path.basename(zip_filename)}', f)
        logger.info(f'文件上传完成: {zip_filename}')

        # 验证文件大小
        remote_size = ftp.size(os.path.basename(zip_filename))
        local_size = os.path.getsize(zip_filename)
        logger.debug(f'远程文件大小: {remote_size} 字节，本地文件大小: {local_size} 字节')

        if remote_size != local_size:
            logger.warning(f'文件大小不匹配! 远程: {remote_size} 本地: {local_size}')

        # 列出当前目录内容
        logger.debug('当前FTP目录内容:')
        ftp.retrlines('LIST', lambda x: logger.debug(x))

        # 关闭连接
        ftp.quit()
        logger.info(f'FTP连接已关闭')
        return True
    except ftplib.all_errors as e:
        logger.error(f'FTP操作失败: {str(e)}')
        return False
    finally:
        if 'ftp' in locals():
            try:
                ftp.quit()
            except:
                pass

def cleanup_local(zip_filename):
    """上传成功后删除本地ZIP文件"""
    try:
        os.remove(zip_filename)
        logger.info(f'已清理本地临时文件: {zip_filename}')
    except Exception as e:
        logger.error(f'清理本地文件失败: {str(e)}')

def main():
    args = parse_arguments()

    # 初始化日志
    # 获取当前日期和时间
    today = datetime.now()
    # weekday()返回0-6，0代表周一，6代表周日,人为加1保持跟周几一致
    weekday_num = today.weekday() + 1
    if args.log_file == "":
        log_path = f"log_back_{weekday_num}.log"
    else:
        log_path = f"{args.log_file}_{weekday_num}.log"
    global logger
    # rotate_logs(log_path, args.max_log_size)
    logger = setup_logging(log_path)
    clear_log_file(log_path)
    logger.info('='*50)
    logger.info('开始日志备份任务')
    logger.info(f'参数: {vars(args)}')

    # 检查源目录是否存在
    if not os.path.isdir(args.source):
        logger.error(f'源目录不存在: {args.source}')
        return

    # 创建带时间戳的ZIP文件名
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    zip_filename = f"{args.zip_prefix}_{timestamp}.log.zip"

    logger.info(f'开始打包日志: {args.source}')
    if not create_zip(zip_filename, args.source, args.keep_days, args.extensions):
        return

    logger.info(f'开始FTP上传到: {args.ftp_host}')
    if upload_via_ftp(zip_filename, args.ftp_host, args.ftp_user, args.ftp_pass,
                     args.ftp_dir, args.debug_ftp):
        cleanup_local(zip_filename)
        logger.info('备份任务完成!')
    else:
        logger.error('备份任务完成，但有错误发生。')

if __name__ == '__main__':
    main()# 获取当前日期和时间
