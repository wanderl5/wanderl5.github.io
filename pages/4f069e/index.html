# awk

# awk内置参数

$n 当前记录的第n个字段，比如n为1表示第一个字段，n为2表示第二个字段。  
$0 这个变量包含执行过程中当前行的文本内容。  
[N] ARGC 命令行参数的数目。  
[G] ARGIND 命令行中当前文件的位置（从0开始算）。  
[N] ARGV 包含命令行参数的数组。  
[G] CONVFMT 数字转换格式（默认值为%.6g）。  
[P] ENVIRON 环境变量关联数组。  
[N] ERRNO 最后一个系统错误的描述。  
[G] FIELDWIDTHS 字段宽度列表（用空格键分隔）。  
[A] FILENAME 当前输入文件的名。  
[P] FNR 同NR，但相对于当前文件。  
[A] FS 字段分隔符（默认是任何空格）。  
[G] IGNORECASE 如果为真，则进行忽略大小写的匹配。  
[A] NF 表示字段数，在执行过程中对应于当前的字段数。  
[A] NR 表示记录数，在执行过程中对应于当前的行号。  
[A] OFMT 数字的输出格式（默认值是%.6g）。  
[A] OFS 输出字段分隔符（默认值是一个空格）。  
[A] ORS 输出记录分隔符（默认值是一个换行符）。  
[A] RS 记录分隔符（默认是一个换行符）。  
[N] RSTART 由match函数所匹配的字符串的第一个位置。  
[N] RLENGTH 由match函数所匹配的字符串的长度。  
[N] SUBSEP 数组下标分隔符（默认值是34）。

BEGIN {
    # 从命令行获取目标时间戳
    beg_time = ARGV[1];
    end_time = ARGV[2];

    # 提取日志中的时间戳（假设时间戳在第 1 列的前 8 个字符）
    log_time = substr($1, 1, 8);
    # print(log_time, end_time)
    # 判断时间戳是否在目标范围内
    if (log_time >= beg_time && log_time <= end_time) {
        print $0;  # 输出符合条件的日志行
    }
}

# 即期tps统计(tps_spot.awk)

# tps_spot.awk
BEGIN {
    # 从命令行获取关键字
    keyword = ARGV[1];
    # 清除 ARGV[1]，避免被当作输入文件
    ARGV[1] = "";
}
$0 ~ keyword { # 匹配模式
    split($1, time, ":");  # 提取时间部分，按 ":" 分割
    second = time[1] ":" time[2] ":" time[3];  # 组合成完整的时间（精确到秒）
    tps[second]++;  # 统计每秒的 TPS
    total_tps++;  # 统计总 TPS
}
END {
    # 将时间戳和 TPS 存储到数组中
    count = 0;
    for (sec in tps) {
        timestamps[count] = sec;  # 存储时间戳
        tps_values[count] = tps[sec];  # 存储 TPS
        count++;
    }

    # 对时间戳进行排序
    n = asort(timestamps);  # 使用 asort 对时间戳排序

    # 输出每秒的 TPS
    for (i = 1; i <= n; i++) {
        print "Time:", timestamps[i], "TPS:", tps[timestamps[i]];
    }

    # 计算并输出总 TPS 和平均 TPS
    if (n > 0) {
        avg_tps = total_tps / n;  # 计算平均 TPS
        print "\nTotal TPS:", total_tps;
        print "Average TPS:", avg_tps;
    } else {
        print "\nNo data found.";
    }
}

# 响应时间统计
BEGIN{
c1=0
c2=0
c3=0
sum=0
count=0
FS=","
}
{
	sum+=$3
	count+=1
	a[count]=$3
	if($3>1200000) c1+=1
	if($3>400000) c2+=1
	if($3>600000) c3+=1
}
END{
lin90=int(0.9*count)
print "MIN is: "a[1]/1000 "ms"
print "90% is: "a[lin90]/1000 "ms"
print "MAX is: "a[count]/1000 "ms"
print "AVG is:  "(sum/count)/1000
print "More than 1200 ms line num is:" c1";Percent is:"(c1/count)*100"%"
print "More than 400 ms line num is:" c2";Percent is:"(c2/count)*100"%"
print "More than 600 ms line num is:" c3";Percent is:"(c3/count)*100"%"
}

# 商品tps统计(tps_com.awk)

# tps_com.awk
/SendComFeedPriceToService OK/ {  # 只处理包含 "SendComFeedPriceToService OK" 的行
    split($1, time, ":");  # 提取时间部分，按 ":" 分割
    second = time[1] ":" time[2] ":" time[3];  # 组合成完整的时间（精确到秒）

    # 提取 size|3| 中的数字
    if (match($0, /size\|[0-9]+\|/)) {
        size = substr($0, RSTART + 5, RLENGTH - 6);  # 提取数字部分
        tps[second] += size;  # 累加每秒的总次数
        total_tps += size;  # 累加总次数
    }
}
END {
    # 将时间戳和 TPS 存储到数组中
    count = 0;
    for (sec in tps) {
        timestamps[count] = sec;  # 存储时间戳
        tps_values[count] = tps[sec];  # 存储 TPS
        count++;
    }

    # 对时间戳进行排序
    n = asort(timestamps);  # 使用 asort 对时间戳排序

    # 输出每秒的 TPS
    for (i = 1; i <= n; i++) {
        print "Time:", timestamps[i], "TPS:", tps[timestamps[i]];
    }

    # 计算并输出总 TPS 和平均 TPS
    if (n > 0) {
        avg_tps = total_tps / n;  # 计算平均 TPS
        print "\nTotal:", total_tps;
        printf "Average TPS: %.2f\n", avg_tps;  # 输出平均 TPS，保留两位小数
    } else {
        print "\nNo data found.";
    }
}
