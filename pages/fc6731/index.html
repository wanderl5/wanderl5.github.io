# 条件渲染1:v-if/else

// 控制是否显示: show=false时,这个span标签将不会出现在实际的html代码中;仅留下 " <!--v-if--> "
<span v-if="show">helloow world1: {{content}} </span>

// 条件判断
<script>
<template>
    <button @click="handleClick" id="btn1" :title="inputValue">add</button>
    <div>num:{{num}}</div>
    <span v-if="isEQ1">isEQ1</span>
    <span v-else-if="isEQ2">isEQ2</span>
    <span v-else>else:{{ num }}</span>
</template>


data() {
return {
    num: 0,
    isEQ1: false,
    isEQ2: false,
    isElse: false,
};
},
methods: {
handleClick() {
    this.num += 1;
    this.isEQ1 = this.isEQ2 = this.isEQ3 = false;
    if (this.num == 1) this.isEQ1 = true;
    if (this.num == 2) this.isEQ2 = true;
    if (this.num >= 3) this.isElse = false;
}
}, 

</script>

# 数据双向绑定 v-model

<script>
    app = Vue.createApp({
        data(){
            return {
                inputValue : '',
                list: ['hello', 'world', 'dell']
            }
        },
        methods:{
            handleClick(){
                this.list.push(this.inputValue); // this 必不可少
                this.inputValue = ""; // v-model为双向绑定,赋值后,input框同步清空
            }
        },
        template: `
        <div>
            <input v-model="inputValue"/>
            <button v-on:click="handleClick" id='btn1'>add</button>
            <ul>
                <li v-for="(item,index) of list">{{index}} - {{item}} </li>
            </ul>

        </div>
       `
    });
    app.mount('#root')
</script>

# 动态事件绑定设置

初始为点击事件，点击后变成鼠标进入事件触发;

<template>
  <div @[event]="handleEvent">event: {{ msg }} ,</div>
</template>
  
  <script>
export default {
  name: "App",
  data() {
    return {
      num: 0,
      msg: "hello",
      event: "click",
      eventlist: ["click", "mouseenter"],
    };
  },
  methods: {
    handleEvent() {
      this.num += 1;
      let idx = this.num % 2;      
      let curEvent = this.event;
      this.event = this.eventlist[idx];
      console.log("handleEvent, changed event from ", curEvent, " -> " , this.event);
    },
  },
};
</script>

# computed属性

相当于是动态只读属性,在计算因素变化时会触发计算

<template>
  <div>
    {{ total }}
    <!-- <button @click="click">click</button> -->
  </div>

</template>
  
<script>

export default {
  name: "App",
  data() {
    return {
      // total: 0, // 不用定义,定义了则会报重复定义
      price: 3,
      amount: 10,
    }
  },
  computed: {
      // 相当于新增一个total属性
      total() { 
        // return parseInt(this.price, 10) * parseInt(this.amount, 10)
        return this.price * this.amount
      }
  },
  methods: {
    // click(){

    // },
  },
};
</script>

# compute属性和get方法的区别

每次界面渲染，都会重新计算

<template>
  <div>
    method_{{ msg }}: {{ getTotal() }} <br/>
    computed_{{ msg }}: {{ total }} <br/>
    <button @click="click">click</button>
  </div>

</template>
  
<script>

export default {
  name: "App",
  data() {
    return {
      price: 5,
      amount: 10,
      index: 0,
      msg: "hello",
    }
  },
  computed: {
      // 相当于新增一个total属性
      total() { 
        return this.price * this.amount + ':' + Date.now();
      }
  },
  methods: {
    getTotal(){
      return this.price * this.amount + ':' + Date.now();
    },
    click (){
      this.index += 1;
      this.msg = 'cy' + this.index;
    }
  },
};
</script>

# watch

  • 监视指定属性，当产生变化时触发
<template>
  <div>
    price: {{ price }}, amount: {{ amount }} <br/>
    method_{{ msg }}: {{ getTotal() }} <br/>
    computed_{{ msg }}: {{ total }} <br/>
    watch_{{ msg }}: {{ watchTotal }} <br/>
    <button @click="click">click</button>
  </div>

</template>
  
<script>
export default {
  name: "App",
  data() {
    return {
      price: 5,
      amount: 10,
      index: 0,
      msg: "hello",
      watchTotal: 0,
    }
  },
  computed: {
      total() { 
        return this.price * this.amount + ':' + Date.now();
      }
  },
  methods: {
    getTotal(){
      return this.price * this.amount + ':' + Date.now();
    },
    click (){
      this.price += 1;
    }
  },
  watch: {
    // 当price变化时，函数会执行，跟computed属性有相似性
    // 当前值和之前的值入参固定顺序:current,prev,但具体参数名称可以自定义,甚至于参数也可以不要
    // price(current, prev1){
    //   console.log("prev:", prev1, "current:", current)
    // }
    price(){
      this.watchTotal = this.price * this.amount + 20;
      console.log(this.watchTotal)
    }
  }
};
</script>
