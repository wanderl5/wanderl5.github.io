# c++98的时间处理相关结构

# time_t

  • 这本质上是一个长整数，表示从1970-01-01 00:00:00到目前计时时间的秒数，如果需要
  • 仅精确到 秒

# timeval

  • 在头文件<time.h>
  • tv_usec精确到 微秒
struct timeval {  
    time_t        tv_sec;     // seconds
    suseconds_t   tv_usec;   // microseconds 微秒
 };

# tm

  • 仅精确到 秒
  • tm_wday对应周几
  • tm_yday对应一年中的第几天
  • tm_year表示从1900年到目前计时时间间隔多少年，如果是手动设置值的话，tm_isdst通常取值-1
struct tm {  
    int tm_sec;     // seconds after the minute - [0,59]  
    int tm_min;     // minutes after the hour - [0,59]    
    int tm_hour;    // hours since midnight - [0,23]      
    int tm_mday;    // day of the month - [1,31]          
    int tm_mon;     // months since January - [0,11]      
    int tm_year;    // years since 1900 
    int tm_wday;    // days since Sunday - [0,6]   
    int tm_yday;    // days since January 1 - [0,365]   
    int tm_isdst;   // daylight savings time flag
};

# epoch秒转换为正常格式(Convert from epoch to human-readable date)

// 获取当前日期时间并转换为正常格式
#include <stdio.h>
#include <time.h>

int main(void)
{
    time_t     now;
    struct tm  ts;
    char       buf[80];

    // Get current time
    time(&now);

    ts = *localtime_r(&now, &ts);
    strftime(buf, sizeof(buf), "%a %Y-%m-%d %H:%M:%S", &ts);
    printf("%s\n", buf);
    return 0;
}

# 获取指定时间的epoch秒数

time_t getEpochSec(int date, int time){
    struct tm t;
    time_t t_of_day;

    int year = date / 10000;
	int month = (date / 100) % 100;
	int day = date % 100;

    int hh = time/10000;
	int mm = time/100%100;
	int ss = time%100;

    t.tm_year = year-1900;  // Year - 1900
    t.tm_mon = month - 1;   // Month, where 0 = jan
    t.tm_mday = day;        // Day of the month
    t.tm_hour = hh;
    t.tm_min = mm;
    t.tm_sec = ss;
    t.tm_isdst = -1;        // Is DST on? 1 = yes, 0 = no, -1 = unknown
    t_of_day = mktime(&t);   
    // printf("[%d %d]seconds since the Epoch: %ld\n", date, time, (long) t_of_day);

    return t_of_day;
}

# epoch转换为可读时间

std::string epochToString(time_t &tt, const std::string &format, int timezone=8){
    char buf[80];
    tt += timezone * 3600; // 默认按GMT8
    // get GMT time
    struct tm *ts = gmtime(&tt);
    // Format time, "ddd yyyy-mm-dd hh:mm:ss zzz"
    strftime(buf, sizeof(buf), format.c_str(), &(*ts));
    return std::string(buf);
}

# 获取当前时间

#include <stdio.h>
#include <time.h>
#include <sys/time.h>
#include <string.h>
using namespace std;

int main(void)
{
    struct timeval tv;
    struct tm st;
    gettimeofday(&tv, NULL);
    time_t t1;
    time(&t1);
    localtime_r(&t1, &st);
    int year = st.tm_year + 1900;
    int month = st.tm_mon + 1;
    int day = st.tm_mday;
    int hour = st.tm_hour;
    int minute = st.tm_min;
    int second = st.tm_sec;
    int usecond = tv.tv_usec;

    char buf[30];
    memset(buf, 0, 30);
    sprintf (buf, "%04d-%02d-%02d %02d:%02d:%02d.%06ld", year, month, day, hour, minute, second, usecond);
    printf("now: %s\n", buf);
    return 0;
}

# 计算时间间隔

  • 秒级： 采用时间戳直接相减
#include <time.h>
#include <stdio.h>
#include <unistd.h>
int main()
{
    time_t t1 = time(NULL);
    printf("t1 is : %ld\n", t1);
    sleep(3);
    time_t t2 = time(NULL);
    printf("t2 is : %ld\n", t2);

    long interval = t2 - t1;
    float interval2 = difftime(t2, t1);
    printf("the time interval is: %ld, %f\n", interval, interval2);
    return 0;
}

  • 微秒/毫秒级
#include <stdio.h>
#include <time.h>
#include <sys/time.h>
#include <unistd.h>
using namespace std;

typedef enum {
    ttSeconds = 0,
    ttMSeconds, // 毫秒  
    ttUSeconds, // 微秒
    ttUEnd
}timeType;
long getTimeInterval(struct timeval &tv, struct timeval &tv2, timeType tt){
    if (tt == ttSeconds){
        return tv2.tv_sec - tv.tv_sec;
    } else if (tt == ttMSeconds){
        return (tv2.tv_sec - tv.tv_sec) * 1000 + (tv2.tv_usec - tv.tv_usec)/1000;
    } else{
        return (tv2.tv_sec - tv.tv_sec) * 1000 * 1000 + (tv2.tv_usec - tv.tv_usec);
    }
}
int main(void)
{
    struct timeval tv;
    gettimeofday(&tv, NULL);

    usleep(300 *  1000);

    struct timeval tv_now;
    gettimeofday(&tv_now, NULL);
    long interval = getTimeInterval(tv, tv_now, ttUSeconds);
    
    printf("now: %ld\n", interval);
    return 0;
}

# 附录1 strftime格式

  • %a 星期几的简写
  • %A 星期几的全称
  • %b 月分的简写
  • %B 月份的全称
  • %c 标准的日期的时间串
  • %C 年份的后两位数字
  • %d 十进制表示的每月的第几天
  • %D 月/天/年
  • %e 在两字符域中，十进制表示的每月的第几天
  • %F 年-月-日
  • %g 年份的后两位数字，使用基于周的年
  • %G 年分，使用基于周的年
  • %h 简写的月份名
  • %H 24小时制的小时
  • %I 12小时制的小时
  • %j 十进制表示的每年的第几天
  • %m 十进制表示的月份
  • %M 十时制表示的分钟数
  • %n 新行符
  • %p 本地的AM或PM的等价显示
  • %r 12小时的时间
  • %R 显示小时和分钟：hh:mm
  • %S 十进制的秒数
  • %t 水平制表符
  • %T 显示时分秒：hh:mm:ss
  • %u 每周的第几天，星期一为第一天 （值从0到6，星期一为0）
  • %U 第年的第几周，把星期日做为第一天（值从0到53）
  • %V 每年的第几周，使用基于周的年
  • %w 十进制表示的星期几（值从0到6，星期天为0）
  • %W 每年的第几周，把星期一做为第一天（值从0到53）
  • %x 标准的日期串
  • %X 标准的时间串
  • %y 不带世纪的十进制年份（值从0到99）
  • %Y 带世纪部分的十进制年份
  • %z，%Z 时区名称，如果不能得到时区名称则返回空字符。
  • %% 百分号
