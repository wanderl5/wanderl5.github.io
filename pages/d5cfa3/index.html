# ref处理基础类型

<script>
    // 响应式引用 ref, reactive
    // 原理: 通过proxy对数据进行封装，当数据变化时，触发模板等内容的更新
    // ref 处理 基础类型 的数据
    const app = Vue.createApp({
        setup(props, context) {
            const { ref } = Vue;
            // 'dell'变成proxy({value: 'dell'}) - 响应式引用
            let name = ref('dell');
            setTimeout(() => {
                name.value = 'lee';
                console.log(name); // 从控制台可以看到name已经变了,但展示并未变,即未响应式绑定
            }, 2000);
            handleClick = () => { alert('handleClick')};
            return {
                name,
                handleClick,
            }                
        },
        // ref类型的字段使用时底层会自动取其value
        template: ` <div @click="handleClick"> name: {{name}}</div>  `
    });
    const vm = app.mount('#root');
</script>

# readonly只读保护

<script>
    // 响应式引用 reactive
    const app = Vue.createApp({
        setup(props, context) {
            const { reactive, readonly } = Vue;
            // 'dell'变成proxy({value: 'dell'}) - 响应式引用
            let nameObj = reactive({ name: 'dell'});
            let readOnlyNameObj = readonly({ name: 'dell'});
            setTimeout(() => {
                nameObj.name = 'lee';
                readOnlyNameObj.name = 'lee';
                console.log(nameObj.name); // 从控制台可以看到name已经变了,但展示并未变,即未响应式绑定
            }, 2000);
            return {
                nameObj,
                readOnlyNameObj,
            }                
        },
        // ref类型的字段使用时底层会自动取其value
        template: ` <div> name: {{nameObj.name}}, readonlyName: {{readOnlyNameObj.name}} </div>  `
    });
    const vm = app.mount('#root');
</script>

# toRefs通过解构转换为proxy

<script>
    // 响应式引用 reactive
    const app = Vue.createApp({
        setup(props, context) {
            const { reactive, toRefs } = Vue;
            let nameObj = reactive({ name: 'dell', age: 28 });
            setTimeout(() => {
                nameObj.name = 'lee';
                console.log(nameObj.name); // 从控制台可以看到name已经变了,但展示并未变,即未响应式绑定
            }, 2000);
            // name: proxy({name: 'dell'})
            // age: proxy({age: 28})
            const { name, age } = toRefs(nameObj);
            return { name, age, }
        },
        // ref类型的字段使用时底层会自动取其value
        template: ` <div> name: {{name}}, age: {{age}} </div>  `
    });
    const vm = app.mount('#root');
</script>

# toRef

  • 可解决toRefs空对象无法赋值问题
<script>
    // 响应式引用 reactive - 错误示范, 需用toRef解决
    const app = Vue.createApp({
        setup(props, context) {
            const { reactive, toRef } = Vue;
            const data = reactive({ age: 18 });
            setTimeout(() => {
                data.age = 20;
                data.sex = 'male';
            }, 2000);
            return {
                age: toRef(data, 'age'),
                sex: toRef(data, 'sex'), // 对象为空时,默认按空对象设置,后续仍可赋值
            } // 导出对象
        },
        template: ` <div> age: {{age}}, sex: {{sex}} </div>  `
    });
    const vm = app.mount('#root');
</script>

# context: attrs,slots,emit

  • attrs: 存储None-Props属性
  • slots: 父组件内容slot
  • emit: 传递事件到父组件
<script>
    // 响应式引用 reactive - 错误示范, 需用toRef解决
    const app = Vue.createApp({
        methods: {
            handleChange() {
                alert('change in app');
            },
        },
        template:`<child @change="handleChange" attr='attr1'>parent</child>`
    });
    app.component('child', {
        // props: ['attr'], // 如果定义了props,则attrs中无法展示，因属性已非None-Props属性
        setup(props, context) {
            const { h } = Vue;
            const { attrs, slots, emit } = context;
            function handleClick() {
                emit('change'); // 触发父组件事件
            }
            console.log(attrs.attr); // attrs可展示None-Props属性
            console.log(slots.default()); // 能获取到子组件的slots,但console展示不太好看,可通过虚拟DOM返回到界面来看
            // return { () => h('div', {}, slots.default() }
            return { mydata: slots.default(),
                handleClick,
            }
        },
        template: ` <div @click="handleClick"> <slot>child</slot> </div>  `
    });
    const vm = app.mount('#root');
</script>

# TodoList实例

# 简单实例

<script>
    const app = Vue.createApp({
        setup(props, context) {
            const { ref, reactive } = Vue;
            const inputValue = ref('请输入后press enter');
            const datas = reactive([]);

            // input事件在每次值改变时触发
            const handleInputChange = (e) => {
                inputValue.value = e.target.value;  // input控件的值
                console.log("input change:", inputValue.value);
            }
            const handleFocus = (e) => {
                inputValue.value=''; // 焦点到后,控件清空
            }
            const handleSubmit = () => { // 提交或enter触发
                if (inputValue.value.length != 0){
                    datas.push(inputValue.value);
                    inputValue.value=''; // 输入完成后,控件清空
                } else{
                    console.log("no submit needed,empty:", inputValue.value,";");
                }
            }
            return {
                inputValue,
                datas,
                handleInputChange,
                handleFocus,
                handleSubmit,
            }
        },
        template: ` <div>
                <input :value="inputValue" @input="handleInputChange" @focus="handleFocus" @keydown.enter="handleSubmit"/>
                <button @click="handleSubmit">add</button>
                <ul>
                    <li v-for="(item, index) in datas" :key="index">{{item}}</li>
                </ul>
             </div>`
    });
    const vm = app.mount('#root');
</script>

# 封装后实例

<script>
    // 关于list操作进行了封装
    const listRelativeEffect = () => {
        const { reactive } = Vue;
        const datas = reactive([]);
        const addItemToList = (item) => {
            if (item.length != 0) {
                datas.push(item);
                item = ''; // 输入完成后,控件清空
            } else {
                console.log("no submit needed,empty:", item, ";");
            }
        }
        return {
            datas, addItemToList,
        }
    }

    // 关于input的操作进行了封装
    const inputRelativeEffect = () => {
        const { ref } = Vue;
        const inputValue = ref('请输入');
        const handleFocus = (e) => {
            inputValue.value = ''; // 焦点到后,控件清空
        }
        // input事件在每次值改变时触发
        const handleInputChange = (e) => {
            inputValue.value = e.target.value;  // input控件的值
            console.log("input change:", inputValue.value);
        }
        return {
            inputValue, handleFocus,handleInputChange,
        }
    }

    const app = Vue.createApp({
        setup(props, context) { // 流程中转
            const { datas, addItemToList } = listRelativeEffect();
            const { inputValue, handleFocus, handleInputChange } = inputRelativeEffect();
            return { datas, addItemToList, inputValue, handleFocus, handleInputChange,}
        },
        template: ` <div>
                <input :value="inputValue" @focus="handleFocus" @input="handleInputChange" @keydown.enter="addItemToList"/>
                <button @click="() => addItemToList(inputValue)">add</button>
                <ul>
                    <li v-for="(item, index) in datas" :key="index">{{item}}</li>
                </ul>
             </div>`
    });
    const vm = app.mount('#root');
</script>

# watch 监听器

  • 需指定监听对象
  • 执行有惰性lazy,仅在监听对象变化时执行 （如果想变成非惰性,可以显式指定immediate: true）
  • 可获取到变动前后的值

# 单属性监听

<script>
    const app = Vue.createApp({
        setup(props, context) {
            const {ref, watch } = Vue;
            const name = ref('dell');
            // 参数顺序固化
            watch(name, (cur, pre) => {
                console.log(cur, pre);
            })
            return { name }
        },
        template: ` <div>
            <input v-model="name"/><br/>
            name is {{name}}
            </div>`
    });
    const vm = app.mount('#root');
</script>

# 多属性监听

<script>
    const app = Vue.createApp({

# watchEffect

  • 不需传递监控对象,自动感知代码依赖
  • 无惰性,函数定义后立即执行
  • 不能获取之前数据的值
<script>
    const app = Vue.createApp({
        setup(props, context) {
            const { reactive, watch, watchEffect, toRefs } = Vue;
            const nameObj = reactive({ name: '大耳朵', eng: 'dell' });
            // 有一定惰性
            watch([() => nameObj.name, () => nameObj.eng], ([cur, cureng], [pre, preeng]) => {
                console.log("name:", pre, '->', cur, "; eng:", preeng, '->', cureng);
            })
            // 没有惰性,立即执行
            const stop = watchEffect(()=>{
                console.log("watchEffect"); // 仅在初始化时运行一次,后续都不会运行,原因是watchEffect仅在代码关联对象时执行
                console.log("watchEffect:",nameObj.name);
                setTimeout(() => {
                    stop(); // 再次调用自己,会触发停止监听,该方式watch和watchEffect都适用
                    console.log("watchEffect stoped");
                }, 5000);
            })
            const { name, eng } = toRefs(nameObj);
            return { name, eng }
        },
        template: ` <div>
            <input v-model="name"/><br/>
            name is {{name}} <br/>
            <input v-model="eng"/><br/>
            engname is {{eng}}
            </div>`
    });
    const vm = app.mount('#root');
</script>

# provide,inject

<script>
    const app = Vue.createApp({
        setup(props, context) {
            const { ref, provide, readonly } = Vue;
            const name = ref('dell');
            // 强制要求子类inject得到的是只读对象
            provide('name', readonly(name)); 
            // 保证单向数据流,父组件的对象仅能由父组件修改
            provide('changeName', (value) =>{
                name.value = value;
            })
            return {name,}
        },
        template: ` <child/>`
    });
    app.component('child', {
        setup(props, context) {
            const { inject } = Vue;
            const name = inject('name');
            const changeName = inject('changeName');
            const handleClick = () => {
                // name.value = 'lee'; // 错误写法,会因为readonly属性而无法修改
                changeName('lee');
            }
            return {name, handleClick}
        },
        template: ` <div @click="handleClick">{{name}}</div>`
    })
    const vm = app.mount('#root');
</script>

# setup语法糖

setup是Vue3.0后推出的语法糖，并且在Vue3.2版本进行了大更新，像写普通JS一样写vue组件，对于开发者更加友好了；按需引入computed、watch、directive等选项，一个业务逻辑可以集中编写在一起，让代码更加简洁便于浏览

  • Comp1.vue
<script setup lang="ts">
import { ref, reactive, computed } from 'vue'    
// 普通变量
const msg = 'Hello!'
 
// 响应式变量
let num = ref(1111)         // ref声明基本类型变量
const obj = reactive({        // reactive声明对象类型变量，如Object、Array、Date...
    key: 'this is a object'
})
 
// 函数
function log() {
    console.log(msg)          // Hello
    console.log(num.value)    // 1111（可根据input输入值而改变）
    console.log(obj.key)      // this is a object
}
let count = ref(0)
const countPlus = computed(()=>{
    return count.value+1
})

function handleClick() {
    num.value += 1;
}
</script>
 
<template>
    <h1>{{ msg }}</h1>
    <p>{{obj.key}}</p>
    <input v-model="num" type="text" />
    <button @click="log">打印日志</button>
    <h1 @click="handleClick">计数：{{ countPlus }}, num:{{ num }}</h1>
</template>
