# 重载new/delete实现泄漏点监控

  • 在程序中加入new,delete的重载,将内存泄漏记录到日志中,并在new时将堆栈stack记录下来与地址关联上
  • 根据new,delete配对情况,定位出未delete的内存,从而定位泄漏点

void recordTrace(void* memory, FILE* fp) {

}

# 内存泄漏检查程序 - memoParser

  • 读取cdemo1产生的内存文件,根据正则表达式捕获到new,delete的长度,地址,堆栈等信息
  • 按地址检查是否有delete,如果delete则删除,直到最后退出时,将内存中存在泄漏的点,使用addr2line解释堆栈
#include <execinfo.h> // 可能需要此头文件，取决于libbacktrace的具体实现
#include "backtrace.h" // libbacktrace提供的头文件
#include <regex>
# include <iostream>
# include <fstream>
#include <cstring>
#include <unordered_map>
#include <set>

using namespace std;
const char VERSION[] = "01.04.14.01";
TSimpleLog* logger;

enum class MemoAction { New, Delete };

// 内存清理控制
class TMemoClean{
public:
    MemoAction memoAction;
    unsigned long size; // 内存大小
    char address[20];
    char stack[500];
};

struct TLeakCount{
public:
    unsigned long size;     // 总泄漏长度,单位byte
    unsigned long count;    // 总泄漏次数
    char stack[500];
};

void parseStack(const basic_string<char> &stak, string basicString);
std::vector<std::string> splitByChar(const std::string& input, char delimiter) ;
std::string getLastPart(const std::string& input, char delimiter);
void replaceString(std::string &str,std::string target, std::string replacement);

int main(int argc, char** argv)
{
    if (argc != 3){
        cout << "Usage: ../bin/memoParser memo.log ../bin/cdemo1" << endl;
        return -1;
    }
    string app = string(argv[2]);
    string appName = getLastPart(app, '/');
    logger = new TSimpleLog("memoParser_" + appName + ".log");
    logger->setLogLevel(LogLevel::DEBUG);
    logger->log(LogLevel::DEBUG, VERSION);

    // 读取文件
    // new,delete按顺序放入map,找到直接remove
    // 剩余内容根据堆栈去重,得到泄漏点
    std::ifstream inputFile(argv[1]);
    if (!inputFile) {
        std::cerr << "Failed to open file.\n";
        return 1;
    }
//===delete [4] bytes at address[0x55965b728090]  ...
//===new [4] bytes at address[0x55965b728090] ,BT{../bin/cdemo1(+0x124e) [0x55965a9b224e],../bin/cdemo1(_Znwm+0x98) [0x55965a9b23a1]}
    std::string line;
    unordered_map<std::string , TMemoClean> memoCleanMap;
    while (std::getline(inputFile, line)) { // 读取一行到line变量
//        std::cout << "Line: " << line << '\n'; // 处理或打印当前行
        TMemoClean memoCount;
        if (line.find("===new ")!=string::npos){
            memoCount.memoAction = MemoAction::New;
            std::regex e(".*\\[(.+)\\] bytes at address\\[(.+)\\].+BT\\{(.+)\\}");
            std::smatch m;
            bool found = regex_search(line, m, e);
            if (!found) return 2;
            for (unsigned long i = 0; i < m.size(); ++i) {
                if (i == 1){
                    memoCount.size = atoi(m[i].str().c_str());
                } else if (i == 2){
                    memset(memoCount.address, 0, sizeof(memoCount.address));
                    memcpy(memoCount.address, m[i].str().c_str(), sizeof(memoCount.address));
                } else if (i == 3){
                    memset(memoCount.stack, 0, sizeof(memoCount.address));
                    memcpy(memoCount.stack, m[i].str().c_str(), sizeof(memoCount.stack));
                }
            }
            memoCleanMap.emplace(memoCount.address, memoCount);
        } else {
            memoCount.memoAction = MemoAction::Delete;
            memoCount.size = 0;
            // ===delete memory at address[0x55965b728090]  ...
            std::regex e("delete memory at address\\[(.+)\\]");
            std::smatch m;
            bool found = regex_search(line, m, e);
//            cout << "m.size:" << m.size() << endl;
            for (unsigned long i = 0; i < m.size(); ++i) {
                if (i == 1){
                    memset(memoCount.address, 0, sizeof(memoCount.address));
                    memcpy(memoCount.address, m[i].str().c_str(), sizeof(memoCount.address));
                }
            }
            if (!found){
                char err[sizeof (memoCount.address) + 30] = {0};
                sprintf(err, "error: address[%s] not found!",  memoCount.address);
                logger->log(LogLevel::ERROR, err);
            }

            auto iter = memoCleanMap.find(memoCount.address);
            if (iter != memoCleanMap.end()) {
                memoCleanMap.erase(iter);
            }
        }
    }
    inputFile.close(); // 关闭文件流

    // 将map放入set去重
    unordered_map<std::string, TLeakCount> memoLeakMap;
    unordered_map<std::string, TMemoClean>::iterator it;
    for (it = memoCleanMap.begin(); it != memoCleanMap.end(); it++){
        TMemoClean clean = it->second;
        auto iter = memoLeakMap.find(clean.stack);
        if (iter != memoLeakMap.end()) {
            iter->second.count ++;
            iter->second.size += clean.size;
        } else{
            TLeakCount leakCount;
            leakCount.size = clean.size;
            leakCount.count = 1;
            memset(leakCount.stack, 0 , sizeof(leakCount.stack));
            memcpy(leakCount.stack, clean.stack, sizeof(leakCount.stack));
            memoLeakMap.emplace(leakCount.stack, leakCount);
        }
    }

    // 根据堆栈调用addr2line打印堆栈
    int totalLeak = memoLeakMap.size();
    if (memoLeakMap.size() != 0){
        logger->log(LogLevel::ERROR, "memory leak found,count:" + std::to_string(totalLeak));
    }
    int loop = 0;
    for (const auto &item : memoLeakMap){
        string stack = item.first;
        TLeakCount leakCount = item.second;
        char log[1024] = {0};
        sprintf(log, "====== [%d/%d]memory leak count[%lu],size[%lu]",
                loop + 1, totalLeak, leakCount.count, leakCount.size);
        logger->log(LogLevel::DEBUG, log);

        string itemWithReturn = leakCount.stack;
        replaceString(itemWithReturn, ",", "\n");
        logger->log(LogLevel::DEBUG, itemWithReturn);
        // 处理堆栈
        parseStack(leakCount.stack, app);
        loop ++;
    }

    logger->log(LogLevel::DEBUG, "all done");
}
void replaceString(std::string &str,std::string target, std::string replacement){
    size_t pos = 0;
    while ((pos = str.find(target, pos)) != std::string::npos) {
        str.replace(pos, target.length(), replacement);
        pos += replacement.length();  // 移动到替换后的新位置，以便查找下一个实例
    }
}
std::string getLastPart(const std::string& input, char delimiter)
{
    vector<std::string> lst = splitByChar(input, delimiter);
    return *lst.rbegin();
}
std::vector<std::string> splitByChar(const std::string& input, char delimiter) {
    std::vector<std::string> result;
    std::string::size_type start = 0;
    auto pos = input.find(delimiter);

    while (pos != std::string::npos) {
        result.push_back(input.substr(start, pos - start));
        start = pos + 1;
        pos = input.find(delimiter, start);
    }

    // 添加最后一个子串（可能没有分隔符）
    result.push_back(input.substr(start));
    return result;
}

void parseStack(const basic_string<char> &stak, string app) {
    // /home/cy/hdd/gitee/cpro/bin/cdemo1(+0x124e) [0x55900340624e],/home/cy/hdd/gitee/cpro/bin/cdemo1(_Znwm+0x98) [0x5590034063a1],/home/c    y/hdd/gitee/cpro/bin/cdemo1(+0x1403) [0x559003406403],/home/cy/hdd/gitee/cpro/bin/cdemo1(+0x1498) [0x559003406498],/lib/x86_64-linux-gnu/libc.so.6(+0x2724a) [0x7fcdbf64624a],
    // /lib/x86_64-linux-gnu/libc.so.6(__libc_start_main+0x85) [0x7fcdbf646305],/home/cy/hdd/gitee/cpro/bin/cdemo1(+0x1151) [0x559003406151]
    // 1.截取堆栈
    // 2.addr2line -e ../bin/cdemo1 -f -C +0x138c
    // 按逗号截取
    std::vector<std::string> stackList = splitByChar(stak, ',');
    char log[1024] = {0};
    int loop = 0;
    int i = 1;
    for (const auto &singleStack :stackList ){
        if (i++ <= 2){
            continue; // 跳过抓取自身,malloc
        }
        // /home/cy/hdd/gitee/cpro/bin/cdemo1(+0x124e) [0x55900340624e],
        std::regex e(".+\\((.+)\\)");
        std::smatch m;
        bool found = regex_search(singleStack, m, e);
        if  (!found) continue;
        memset(log, 0, sizeof(log));
        sprintf(log, "[%d/%lu]%s", loop+1, stackList.size(), singleStack.c_str());
        char cmd[500] = {0};
        sprintf(cmd, "addr2line -e %s -f -C %s", app.c_str(), m[1].str().c_str());
        TInnerShell innerShell(cmd);
        logger->log(LogLevel::DEBUG, string(log));
        innerShell.execute();
        logger->log(LogLevel::DEBUG, "cmd:" + innerShell.command()
            + ", result:\n" + innerShell.log());
        logger->log(LogLevel::DEBUG, "-----------------");
        loop++;
    }
}

# innershell

#ifndef MEM_INNERSHELL_H
#define MEM_INNERSHELL_H

#include <string>
#include <unistd.h>
//#include <list>
//#include <sys/stat.h>
//#include "hance/XString.h"

class TInnerShell {
public:
    TInnerShell(const char* _cmd) : m_command(_cmd) {
		m_fp = NULL;
	}
	
	~TInnerShell() {
		if (m_fp) pclose(m_fp);
	}
	bool execute() {
		m_fp = popen(m_command.c_str(), "r");
		if (m_fp == NULL)
			return false;
		char line[1024];
		while (fgets(line, sizeof(line), m_fp)) {
			trimleft(line);
			m_log.append(line);
		}
		return true;
	}

	std::string &log() {
		return m_log;
	}
    std::string &command() {
        return m_command;
    }
private:
	std::string m_command, m_log;
	FILE *m_fp;
    char *trimleft(char *strbuf) {
		int len = strlen(strbuf);
		if (len <= 0)
			return strbuf;
		for (int i = 0; i < len; i ++) {
			if (!isspace(strbuf[i]))
				return &strbuf[i];
			}
		return &strbuf[len];
	}
};

#endif

// 日志级别枚举
enum class LogLevel {
    DEBUG,
    INFO,
    WARNING,
    ERROR

// 日志类
class TSimpleLog {
public:
    // 构造函数，可传入日志文件名（可选）
    TSimpleLog(const std::string& logFileName = "") {
        if (!logFileName.empty()) {
            logFile.open(logFileName, std::ios_base::app);
            if (!logFile.is_open()) {
                std::cerr << "Failed to open log file: " << logFileName << std::endl;
            }
        }
    }

    // 析构函数，关闭日志文件
    ~TSimpleLog() {
        if (logFile.is_open()) {
            logFile.close();
        }
    }

    // 设置日志级别阈值
    void setLogLevel(LogLevel level) {
        logLevelThreshold = level;
    }

    // 记录日志
    void log(LogLevel level, const char* data) ;
    void log(LogLevel level, std::string data) ;

private:
    // 格式化日志消息
    std::string formatMessage(LogLevel level, const std::string& timestamp, const char* data);

    LogLevel logLevelThreshold = LogLevel::DEBUG;
    std::ofstream logFile;
    std::mutex mutex;
};
#endif //CDEMO_SIMPLELOG_H

#include "SimpleLog.h"

void TSimpleLog::log(LogLevel level, std::string data) {
    log(level, data.c_str());
}
// 记录日志
void TSimpleLog::log(LogLevel level, const char* data) {
    if (level >= logLevelThreshold) {
        std::lock_guard<std::mutex> lock(mutex);

        // 获取当前时间
        std::time_t now = std::time(nullptr);
        char timestampBuffer[32];
        std::strftime(timestampBuffer, sizeof(timestampBuffer), "%Y-%m-%d %H:%M:%S", std::localtime(&now));

        // 格式化日志消息
        std::string message = formatMessage(level, timestampBuffer, data);

        // 输出到控制台
        std::cout << message << std::endl;

        // 输出到文件（如果已打开）
        if (logFile.is_open()) {
            logFile << message << std::endl;
        }
    }
}

// 格式化日志消息
std::string TSimpleLog::formatMessage(LogLevel level, const std::string& timestamp, const char* data) {
    std::string levelName;
    switch (level) {
        case LogLevel::DEBUG:
            levelName = "[DEBUG]";
            break;
        case LogLevel::INFO:
            levelName = "[INFO]";
            break;
        case LogLevel::WARNING:
            levelName = "[WARNING]";
            break;
        case LogLevel::ERROR:
            levelName = "[ERROR]";
            break;
    }

    return timestamp + " " + levelName + " " + std::string(data);
}
