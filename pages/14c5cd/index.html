# 安装Vim-plug

  • 使用wget/curl获取到vim文件
  • 如果无法下载,使用网页打开,另存为plug.vim

# coc.nvim

  • 是一个平台,可以支持很多,此处主要使用其格式化,自动补全

  • 注意切换到taobao, nrm use taobao
  • . / ->均会触发自动跳出提示代码补全界面, tab / shift tab进行上下选择
  • coc.nvim wiki (opens new window)

" Enable syntax highlighting
syntax on

" Set tab width and use spaces instead of tabs
set tabstop=4
set shiftwidth=4
set expandtab

" Enable line numbers
set number
" set nonu

" Disable compatibility mode for better plugin support
set nocompatible

" Enable mouse support in the terminal - may cause can't copy & paste
" set mouse=a

" Improve search behavior
set incsearch
set hlsearch

" Back up files before overwriting
set backup 
set writebackup

let mapleader = ","

""" Plug

inoremap <expr> <cr> coc#pum#visible() ? coc#pum#confirm() : "\<CR>"

使用autocmd根据不同文件类型自动加载指定的vim插件

" 其他初始化配置
augroup myautocmds
    " 清除所有自动命令
    autocmd!

    " 加载C的相关插件
    autocmd FileType cpp setlocal tabstop=4 shiftwidth=4 softtabstop=4

    " 注释的相关插件
    autocmd FileType cpp source ~/.vim/myvim/comment_c.vim
    autocmd FileType c source ~/.vim/myvim/comment_c.vim
    autocmd FileType java source ~/.vim/myvim/comment_c.vim

    autocmd FileType python setlocal tabstop=4 shiftwidth=4 softtabstop=4 nu
    autocmd FileType markdown setlocal textwidth=80
augroup END

" 定义一个函数来使用正则表达式检查当前行是否是注释行
function! IsCommentLine(lineno)
    " 获取当前行的内容
    " let line = getline('.')
    let line = getline(a:lineno)
    
    " 使用正则表达式检查行是否以 // 开头
    if line =~ '^\s*//.*'
        let ret = 1  " 是注释行
    else
        let ret = 0  " 不是注释行
    endif
    "echomsg "current line:" . line . ",IsCommentLine: " . ret
    return ret
endfunction
"unmap <leader>l - 此映射被某插件占用
"noremap <silent> <leader>. :call IsCommentLine('.') <CR>

" 定义一个函数来切换注释状态 - 针对可视化模式下使用,效果也是针对可视化范围
function! ToggleCommentVisual(comment_char)
    " 判断标记的首行和最后行是否以//开头, "."为连接符，ecasape是把特殊字符转义
    if IsCommentLine("'<") == 1 && IsCommentLine("'>") == 1
        // 注意//可能不在行首,此处不能加^
        execute ":'<,'>s/" . escape(a:comment_char, '/') . "//g"
    else
        execute ":'<,'>s/^/" . escape(a:comment_char, '/') . "/g"
    endif
endfunction
function! ToggleCommentNormal(comment_char)
    " 是注释行则取消注释，反之注释
    " if getline(".") =~ '^' . escape(a:comment_char, '/') 
    if IsCommentLine(".") == 1 
        execute ":. s/" . escape(a:comment_char, '/') . "//g"
    else
        execute ":. s/^/" . escape(a:comment_char, '/') . "/g"
    endif
endfunction

" 定义一个映射来调用函数
vnoremap <silent> <leader>. :call ToggleCommentVisual('//') <CR>
noremap  <silent> <leader>. :call ToggleCommentNormal('//') <CR>
