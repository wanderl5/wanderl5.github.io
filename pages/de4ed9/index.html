# cmake结合gtest

# 编译规则文件CMakeLists.txt

cmake_minimum_required(VERSION 2.8)

# 工程名称
project(swap_test)

# 变量定义
set(PATH_TITANPUB /home/emms/workshop/TitanPub_01.01_TitanPub_RELEASE)
#set(PATH_SMDEV /home/emms/workshop/smdev3.1)
set(PATH_GTEST /home/emms/workshop/googletest-master)
set(PATH_APP /home/emms/work/cy/XPRICE_04.04_xprice)

# 源码位置(相对于CMakeLists.txt文件位置的路径)
aux_source_directory(./Extension4Test DIR_Ext_SRCS)
aux_source_directory(./TestCode DIR_SRCS)
aux_source_directory(./include/swap/WorkAction.cpp SWAP_SRCS)

# 指定库文件
# find_library(SM_LIB libsm.a ${PATH_SMDEV}/lib64/linux)
find_library(LANG_LIB liblang.a ${PATH_TITANPUB}/lib64/linux)
find_library(HANCE_LIB libhance.a ${PATH_TITANPUB}/lib64/linux)
find_library(XNET_LIB libxnet.a ${PATH_TITANPUB}/lib64/linux)
find_library(GTEST_LIB libgtest.a ${PATH_GTEST}/build/lib)
find_library(PRICE_LIB libprice.a ${PATH_APP}/lib/lib_linux/oraclelib/oracle12c)
find_library(BASE_LIB libbase.a ${PATH_APP}/lib/lib_linux/oraclelib/oracle12c)
find_library(PESWAP_LIB libpeswap.a ${PATH_APP}/lib/lib_linux/oraclelib/oracle12c)

# 指定找链接库的路径,必须在可执行文件前
link_directories($ENV{ORACLE_HOME}/lib  $ENV{ORACLE_HOME}/precomp/lib)
# 指定编译输出文件
add_executable(swap_test ${DIR_SRCS} ${DIR_Ext_SRCS} ${SWAP_SRCS})
# 指定编译选项
add_definitions(-g -fPIC -std=c++11 -D_DEBUG -D_UNIX -DUNIX -D_PTHREAD_SAFE -D_ORACLE -DORACLE -D_LINUX -DORACA_STORAGE_CLASS=static)

# 指定头文件位置
include_directories(${PATH_TITANPUB}/include)
#include_directories(${PATH_SMDEV}/include)
include_directories(${PATH_GTEST}/googletest/include)
include_directories(${PATH_APP}/include)
include_directories(${PATH_APP}/gtest/xprice/include)
include_directories($ENV{ORACLE_HOME}/precomp/public)
include_directories($ENV{ORACLE_HOME}/rdbms/public)

# 指定链接的lib库
target_link_libraries( swap_test -lclntsh pthread ${GTEST_LIB}  ${BASE_LIB} ${PESWAP_LIB} ${PRICE_LIB} ${SQL_LIB} ${XNET_LIB} ${HANCE_LIB} ${LANG_LIB})

# cmake编译文件解释

# 编译的文件

  • build.make: 编译命令和文件
  • flags.make: 编译的选项
  • link.txt: 程序链接的信息,可直接执行进行链接;
  • CMakeFiles/swap_test.dir/DependInfo.cmake: 可以看到具体的include,cpp,define等信息;

exclude文件

aux_source_directory(${PATH_APP}/include/xnet XNET_SRCS)
list(REMOVE_ITEM XNET_SRCS "${PATH_APP}/include/xnet/xnetHub.cpp")
list(REMOVE_ITEM XNET_SRCS "${PATH_APP}/include/xnet/xnetlog.cpp")
message("cy remove: ${XNET_SRCS}")
