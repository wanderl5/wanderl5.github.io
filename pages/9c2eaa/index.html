IO复用

# 概述

# IO复用三种模型

  • select: 只能管理1024个客户端，且有全量轮循和内存拷贝的问题;
  • poll: 解决了客户端连接问题，但仍有全量轮循和内存拷贝的问题;
  • epoll: 只要内存足够，管理连接没有上限,性能不下降，epoll可以理解为event poll，不同于忙轮询和无差别轮询，epoll会把哪个流发生了怎样的I/O事件通知我们

# select,poll,epool的区别

  • 支持一个进程所能打开的最大连接数
  • select：单个进程所能打开的最大连接数有FD_SETSIZE宏定义，其大小是32个整数的大小（在32位的机器上，大小就是3232，同理64位机器上FD_SETSIZE为3264），当然我们可以对进行修改，然后重新编译内核，但是性能可能会受到影响，这需要进一步的测试。
  • poll：poll本质上和select没有区别，但是它没有最大连接数的限制，原因是它是基于链表来存储的。
  • epoll：虽然连接数有上限，但是很大，1G内存的机器上可以打开10万左右的连接，2G内存的机器可以打开20万左右的连接。
  • FD剧增后带来的IO效率问题
  • select：因为每次调用时都会对连接进行线性遍历，所以随着FD的增加会造成遍历速度慢的“线性下降性能问题”。
  • poll：同上
  • epoll：因为epoll内核中实现是根据每个fd上的callback函数来实现的，只有活跃的socket才会主动调用callback，所以在活跃socket较少的情况下，使用epoll没有前面两者的线性下降的性能问题，但是所有socket都很活跃的情况下，可能会有性能问题；

注意: epool在活跃socket不多的场景下性能会比较高,但在活跃socket不多的情况下，epool有回调函数的性能损耗,select和poll可能性能会优于epool; epool是linux特有,mac和windows都没有

  • 消息传递方式
  • select：内核需要将消息传递到用户空间，都需要内核拷贝动作
  • poll：同上
  • epoll：epoll通过内核和用户空间共享一块内存来实现的（内存拷贝，利用mmap()文件映射内存加速与内核空间的消息传递；即epoll使用mmap减少复制开销）

# 水平触发和边缘触发

  • select和poll只有水平触发
  • epoll同时有水平和边缘触发;

# 水平触发

  • 若事件和数据已在缓冲区,程序调用select时会报告事件,数据不会丢失；
  • 若select已报告事件，但程序没有处理,则下次调用select仍会重新报告;(测试可在事件处理中忽略掉第一个事件)
  • 如果写缓存区没满,写事件始终会被触发;

# 边缘触发

  • socket加入epoll后,如果接收缓冲区不为空,触发可读事件，如果有新的数据达到，再次触发可读事件；
  • epoll触发可读事件后，不管程序是否完成可读事件处理，epoll都不再触发，只有当新的数据达到时才会再次触发;
  • 边缘触发使用 ev.events=EPOLLIN|EPOLLET来表示;
  • socket加入epoll后，若发送缓冲区不为空，触发可写事件；如果发送缓冲区由满变成有空时(注意: 发送缓冲区变化并不能触发写事件)，再次触发可写事件；

# TCP缓冲区

TCP有缓存区,若缓存区已填满,send函数会阻塞;(可以模拟服务端有意usleep) 如果发送端关闭socket,已进入缓存区的数据会继续发送给接收端,由操作系统保证; tcp发送缓冲区默认是2.5M,接收缓冲区1M; 缓冲区补充知识 (opens new window)

int bufsize = 0;
socklen_t optlen = sizeof(bufsize);
// 获取接收缓冲区大小
getsockopt(TcpClient.m_sockfd, SOL_SOCKET, SO_RCVBUF, &bufsize, &optlen);
printf("recv bufsize=%d\n", bufsize);

// 获取发送缓冲区大小
getsockopt(TcpClient.m_sockfd, SOL_SOCKET, SO_SNDBUF, &bufsize, &optlen);
printf("send bufsize=%d\n", bufsize);

# 服务端(select)

#include <thread>
#include <iostream>
#include <memory>
#include <unistd.h>
#include <netdb.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <arpa/inet.h>
#include "mypub/MySocket.h"

// using namespace std;
CTcpServer cTcpServer;

int main(int argc, char *argv[])
{
    if (argc != 2)
    {
        printf("argc:%d \n\n", argc);
        printf("usage: svr port\nExample:\n ./svr 5005 \n\n");
        return -1;
    }

    // 1.创建服务端socket
    cTcpServer.InitServer(atoi(argv[1]));
    int listensock = cTcpServer.m_listenfd;
    printf("listensock=%d\n", listensock);
    if (listensock < 0)
    {
        printf("initserver() failed.\n");
        return -1;
    }

    fd_set readfdset;               // 读事件的集合，包括监听socket和客户端连接上来的socket。
    int maxfd;                      // readfdset中socket的最大值。
    FD_ZERO(&readfdset);            // 初始化读事件socket的集合
    FD_SET(listensock, &readfdset); // 初始化结构体，把listensock添加到集合中。
    maxfd = listensock;             // 记录最大fd

    while (1)
    {
        // 调用select函数时，会改变socket集合的内容，所以要把socket集合保存下来，传一个临时的给select。(也带来一定性能开销)
        fd_set tmpfdset = readfdset;
        // 加入超时控制，10秒无事件则进入超时分支执行;
        struct timeval timeout; timeout.tv_sec = 10; timeout.tv_usec = 0;

        // 事件: 读: 1).新客户连接accept; 2).收到客户端报文 3).客户端连接断开; 写: 4).给客户端发送消息
        // int infds = select(maxfd + 1, &tmpfdset, NULL, NULL, NULL);
        int infds = select(maxfd + 1, &tmpfdset, NULL, NULL, &timeout); // 加入timeout
        // printf("select infds=%d\n",infds);
        // 返回失败。
        if (infds < 0)
        {
            printf("select() failed.\n");
            perror("select()");
            break;
        }
        // 超时，在本程序中，select函数最后一个参数为空，不存在超时的情况，但以下代码还是留着。
        if (infds == 0)
        {
            printf("select() timeout.\n");
            continue;
        }
        // 检查有事情发生的socket，包括监听和客户端连接的socket。
        // 这里是客户端的socket事件，每次都要遍历整个集合，因为可能有多个socket有事件。
        for (int eventfd = 0; eventfd <= maxfd; eventfd++)
        {
            if (FD_ISSET(eventfd, &tmpfdset) <= 0) // 不为1则无事件
                continue;

            if (eventfd == listensock)
            {
                // 如果发生事件的是listensock，表示有新的客户端连上来。
                struct sockaddr_in client;
                socklen_t len = sizeof(client);
                int clientsock = accept(listensock, (struct sockaddr *)&client, &len);
                if (clientsock < 0)
                {
                    printf("accept() failed.\n");
                    continue;
                }
                printf("client(socket=%d) connected ok.\n", clientsock);
                // 把新的客户端socket加入集合。
                FD_SET(clientsock, &readfdset);

                if (maxfd < clientsock)
                    maxfd = clientsock;

                continue;
            }
            else
            {
                // 客户端有数据过来或客户端的socket连接被断开。
                char buffer[1024];
                memset(buffer, 0, sizeof(buffer));

                // 读取客户端的数据。
                ssize_t isize = read(eventfd, buffer, sizeof(buffer));

                // 发生了错误或socket被对方关闭。
                if (isize <= 0)
                {
                    printf("client(eventfd=%d) disconnected.\n", eventfd);
                    close(eventfd); // 关闭客户端的socket。
                    FD_CLR(eventfd, &readfdset); // 从集合中移去客户端的socket。

                    // 重新计算maxfd的值，注意，只有当eventfd==maxfd时才需要计算。
                    if (eventfd == maxfd)
                    {
                        for (int ii = maxfd; ii > 0; ii--)
                        {
                            if (FD_ISSET(ii, &readfdset))
                            {
                                maxfd = ii;
                                break;
                            }
                        }
                        printf("maxfd=%d\n", maxfd);
                    }
                    continue;
                }

                printf("recv(eventfd=%d,size=%d):%s\n", eventfd, isize, buffer);
                // 把收到的报文发回给客户端。
                write(eventfd, buffer, strlen(buffer));
            }
        }
    }

    return 0;
}

# 客户端

/*
 *  程序名：demo47.cpp，此程序演示采用freecplus框架的CTcpClient类实现socket通信的客户端。
 *  作者：C语言技术网(www.freecplus.net) 日期：20190525
*/
#include "mypub/MySocket.h"
 
int main(int argc,char *argv[])
{
  if (argc!=3)
  {
    printf("Using:./cli ip port\nExample:\n./cli 127.0.0.1 5005\n\n"); return -1;
  }

  CTcpClient TcpClient;   // 创建客户端的对象。
 
  if (TcpClient.ConnectToServer(argv[1],atoi(argv[2]))==false) // 向服务端发起连接请求。
  {
    printf("TcpClient.ConnectToServer(\"%s\",%s) failed.\n",argv[1],argv[2]); return -1;
  }
  
  char strbuffer[1024];    // 存放数据的缓冲区。
 
  for (int ii=0;ii<30;ii++)   // 利用循环，与服务端进行5次交互。
  {
    memset(strbuffer,0,sizeof(strbuffer));
    printf("please inpout:");
    scanf("%s", strbuffer);
    printf("发送：%s\n",strbuffer);
    if (send(TcpClient.m_sockfd, strbuffer, strlen(strbuffer), 0) <= 0) break;    // 向服务端发送请求报文。
 
    memset(strbuffer,0,sizeof(strbuffer));
    if (TcpClient.Read(strbuffer,20)==false) break;  // 接收服务端的回应报文。
    printf("接收：%s\n",strbuffer);
 
    sleep(1);
  }
 
  // 程序直接退出，析构函数会释放资源。
}

# select的问题

  • 支持的练技术太小只有1024，且即使调大了,性能也相应会变小;
  • 每次调用select都会把fdset从用户态拷贝到内核,调用后把fdset从内核拷贝到用户态;
  • select返回后,需遍历bitmap,效率低;

# epoll

# 概述

  • 性能卓越,是高性能网络架构的基础(nginx, redis, rpc, libevent);
  • epoll没有内存拷贝,没有轮询,没有遍历;

# 重点描述

  • 创建句柄
int epoll_create(int size);

  • 注册事件
int epoll_ctl(int epfd, int op, int fd, struct epoll_event *event);
1
  • 等待事件
int epoll_wait(int epfd, struct epoll_event *events, int maxevents, int timeout);

# 服务端代码(客户端可复用select模型客户端)

// TODO: ET模式下可能发生的消息错过,通过循环怎么搞定?
#include <thread>
#include <iostream>
#include <memory>
#include <unistd.h>
#include <netdb.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <sys/epoll.h>
#include <arpa/inet.h>
#include "mypub/MySocket.h"
#include <errno.h>

#define MAXEVENTS 100
// using namespace std;
CTcpServer cTcpServer;

int main(int argc, char *argv[])
{
    if (argc != 2)
    {
        printf("argc:%d \n\n", argc);
        printf("usage: svr port\nExample:\n ./svr 5005 \n\n");
        return -1;
    }

    // 1.创建服务端socket
    cTcpServer.InitServer(atoi(argv[1]));
    int listensock = cTcpServer.m_listenfd;
    printf("listensock=%d\n", listensock);
    if (listensock < 0)
    {
        printf("initserver() failed.\n");
        return -1;
    }

    // 设置为非阻塞IO,使select,accept不会阻塞
    /*
    1) 如果在随后调用read函数之前，如果另一个进程已经以写方式打开了该FIFO，并写入了数据，那么正常读取数据。
    2) 如果在随后调用read函数之前，如果另一个进程已经以写方式打开了该FIFO，但是并未写入数据，read调用会立刻出错返回-1， 并设置 errnno为EAGAIN。
    3）如果在随后调用read函数之前，没有其他进程以写方式打开该FIFO，或是曾经有但是在read读时已经关闭了，那么read返回0，表示读到文件结束标志。
    */
    fcntl(listensock, F_SETFL, fcntl(listensock, F_GETFD, 0) | O_NONBLOCK);

    // 创建epoll句柄
    int epollfd = epoll_create(1);

    // 为监听的socket准备刻度事件
    struct epoll_event ev; // 恒明事件的数据结构;
    ev.data.fd = listensock;
    // ev.events = EPOLLIN;   // 读事件
    ev.events = EPOLLIN|EPOLLET;   // 读事件(边缘触发)
    epoll_ctl(epollfd, EPOLL_CTL_ADD, listensock, &ev);
    while (1)
    {
        struct epoll_event evs[MAXEVENTS]; // 存放epoll返回的事件
        // 等待监视的socket有事件发生
        int infds = epoll_wait(epollfd, evs, MAXEVENTS, -1);
        // 返回失败。
        if (infds < 0)
        {
            perror("epoll()");
            break;
        }
        // 超时，在本程序中，select函数最后一个参数为空，不存在超时的情况，但以下代码还是留着。
        if (infds == 0)
        {
            printf("epoll() timeout.\n");
            continue;
        }
        // infds>0表示有事件发生的socket的数量
        // 遍历epoll返回的已发生事件的数组
        for (int ii = 0; ii < infds; ii++)
        {
            printf("loop[%d/%d] [%d]events=%d\n", ii+1, infds, evs[ii].data.fd, ev.events);
            // 如果发生事件的是listensock，表示有新的客户端连上来
            if (evs[ii].data.fd == listensock &&(evs[ii].events & EPOLLIN))
            {
                // test - 故意不处理前两次客户端的连接请求
                // static int ss =0;
                // if (ss++ < 2) continue;
                // while(true){  // 2次无法触发accept,第三次时会一起让3个连接同时连上
                    struct sockaddr_in client;
                    socklen_t len = sizeof(client);
                    int clientsock = accept(listensock, (struct sockaddr *)&client, &len);
                    if (errno == EAGAIN) break; // 未连接上退出循环,否则会持续夯住
                    if (clientsock < 0)
                    {
                        printf("accept() failed.\n");
                        continue;
                    }
                    // 为新客户端准备可读事件,并添加到epoll中
                    memset(&ev, 0, sizeof(struct epoll_event));
                    ev.data.fd = clientsock;
                    ev.events = EPOLLIN;
                    epoll_ctl(epollfd, EPOLL_CTL_ADD, clientsock, &ev); 
                    printf("client[%d] connected ok.\n", clientsock);
                // }
            }
            else
            {
                // TODO: 读取也可能存在读取不完情况;
                // 客户端有数据过来或客户端的socket连接被断开。
                char buffer[1024];
                memset(buffer, 0, sizeof(buffer));
                if(recv(evs[ii].data.fd, buffer, sizeof(buffer), 0) <= 0){
                    // 连接断开
                    printf("client[%d] disconnected.\n", evs[ii].data.fd);
                    close(evs[ii].data.fd);
                }
                else{
                    // 如果客户端有报文发过来
                    printf("recv[%d]:%s\n", evs[ii].data.fd, buffer);
                    // send(evs[ii].data.fd, buffer, strlen(buffer), 0);
                }
            }
        }
    }

    return 0;
}
