int main(int argc, char **argv) {
    if (argc != 2) {
        std::cerr << "usage: ./svr 9999" << std::endl;
        exit(1);
    }
    int port = atoi(argv[1]);
    //evthread_use_pthreads();//enable threads
    init();

    struct sockaddr_in sin;
    memset(&sin, 0, sizeof(struct sockaddr_in));
    sin.sin_family = AF_INET;
    sin.sin_port = htons(port);

    struct event_base *base = event_base_new();
    struct evconnlistener *listener =
            evconnlistener_new_bind(base, listener_cb, base,
                                    LEV_OPT_REUSEABLE | LEV_OPT_CLOSE_ON_FREE, 10,
                                    (struct sockaddr *) &sin, sizeof(struct sockaddr_in));
    std::cout << "rate svr ok at port:" << port << std::endl;
    //事件循环开始
    event_base_dispatch(base);
    evconnlistener_free(listener);
    event_base_free(base);
    return 0;
}

/*
 * 当新客户端连接上服务器，此函数被调用，libevent已经帮助accept了此客户端，该客户端的文件描述符位fd
 */
void listener_cb(struct evconnlistener *listener, evutil_socket_t fd,
                 struct sockaddr *sock, int socklen, void *arg) {
    printf("accept a client %d\n", fd);

    struct event_base *base = (struct event_base *) arg;
    //为此客户端分配一个bufferevent
    struct bufferevent *bev = bufferevent_socket_new(base, fd, BEV_OPT_CLOSE_ON_FREE);
    bufferevent_setcb(bev, socket_read_cb, NULL, socket_event_cb, NULL);
    bufferevent_enable(bev, EV_READ | EV_PERSIST);
    printf("listener_cb ok");

//    char reply[] = "your connection ok";
//    bufferevent_write(bev, reply, strlen(reply));
}

void socket_read_cb(struct bufferevent *bev, void *arg) {
    char msg[BUFFER_SIZE];
    memset(msg, 0, BUFFER_SIZE);
    size_t read_len = bufferevent_read(bev, msg, BUFFER_SIZE - 1);
    msg[read_len] = '\0';
    printf("from client: %s\n", msg);
    int count = atoi(msg);
    {
        IntervalTimer t("rate send", IntervalTimer::ttMS);
        for (long rateNo = 1; rateNo <= count; ++rateNo) {
            std::cout << "==== sending data[" << rateNo << "/" << count << "] ====" << std::endl;
            for (auto iter = RateList.begin(); iter != RateList.end(); iter++) {
                TSRBankBaseItem *pItem = &(*(iter->second));
                pItem->randomRate();
            }
            char total_data[BUFFER_SIZE + 1];
            memset(total_data, 0, BUFFER_SIZE + 1);
            TSRBankBaseItem::packRate(total_data, rateNo, RateList);
            bufferevent_write(bev, total_data, strlen(total_data));
//            std::cout << "send len[" << strlen(total_data) << "]:" << total_data << std::endl;
            std::this_thread::sleep_for(std::chrono::microseconds(10)); // us微秒
        }
    }
}

void socket_event_cb(struct bufferevent *bev, short events, void *arg) {
    if (events & BEV_EVENT_EOF) {
        printf("connection closed\n");
    } else if (events & BEV_EVENT_ERROR) {
        printf("some other error\n");
    }
    //自动关闭套接字和释放读写缓冲区
    bufferevent_free(bev);
}

static void process_message(char *msg, size_t totalLength) {
    size_t pos = 0; // 已读取长度
    if ((msg[0] != '#' && BufferedData.getSize() == 0)
        || (BufferedData.getSize() != 0 && *BufferedData.getData() != '#')) {
        std::cerr << "no begin char, break,pos:" << pos
                  << ":[" << msg + pos << "]" << std::endl;
        std::cout << "buf:" << BufferedData.getData() << std::endl;
        exit(0);
        return;
    }
    while (pos <= totalLength - 1) {
        // 长度不足Head,则直接纳入buffer待后续粘包处理
        if (totalLength - pos < BankRateHeadLen) {
            BufferedData.updateBuffer(msg + pos, totalLength - pos);
            break;
        }
        size_t dataLen = getDataLen(msg + pos);
        if (BufferedData.getSize() == 0) {// 没有遗留的字符串
            if (dataLen + BankRateHeadLen > totalLength - pos) { // 不足整包
                BufferedData.updateBuffer(msg + pos, totalLength - pos);
                break;
            }
            processData(msg, pos);
            pos += dataLen + BankRateHeadLen;
        } else {
            if (dataLen + BankRateHeadLen > totalLength - pos + BufferedData.getSize()) { // 不足整包
                BufferedData.updateBuffer(msg + pos, totalLength - pos);
                break;
            }
            // 记录需要拷贝进去的量
            size_t needLength = dataLen + BankRateHeadLen - BufferedData.getSize();
            processMergeData(msg, pos, needLength);
            pos += needLength;
        }
    }

}

/**
 * 处理Head获取dataLen
 * @param msg 消息
 * @return 返回data长度
 */
static size_t getDataLen(char *msg) {
    char headStr[BankRateHeadLen + 1] = {'\0'};
    size_t buffSize = BufferedData.getSize();
    if (buffSize >= BankRateHeadLen) {
        strncpy(headStr, BufferedData.getData(), BankRateHeadLen);
    } else if (buffSize == 0) {
        strncpy(headStr, msg, BankRateHeadLen);
    } else {
        strncpy(headStr, BufferedData.getData(), buffSize);
        strncpy(headStr + buffSize, msg, BankRateHeadLen - buffSize);
    }
    // 读取head和DataLen
    size_t dataLen = 0;
    // 获取到长度 sscanf读到第一个空格,tab,newline
    if (sscanf(headStr, "#%05zu", &dataLen) != 1) {
        std::cerr << "read dataLen Fail:" << headStr << std::endl;
        throw std::runtime_error("getDataLen fail");
    }
    return dataLen;
}

/**
 * 处理单个通讯包
 * @param msg 数据
 * @param pos 读取起始位置
 */
static void processMergeData(char *msg, size_t pos, size_t needLength) {
    // 合并缓存和剩余长度getNeedLength
    BufferedData.updateBuffer(msg + pos, needLength);
    processData(BufferedData.getData(), 0);
    // 合并的内容
    BufferedData.clear();
}

// 读取事件
static void conn_eventcb(struct bufferevent *bev, short events, void *user_data) {
    if (events & BEV_EVENT_EOF) {
        printf("Connection closed.\n");
    } else if (events & BEV_EVENT_ERROR) {
        printf("Got an error on the connection: %s\n",
               strerror(errno));/*XXX win32*/
    } else if (events & BEV_EVENT_CONNECTED) {
        //连接成功时走这里，并且要客户端第一次触发读事件后连接才真正建立起来
        printf("connect success\n");
        char msg[10] = {'\0'};
        snprintf(msg, sizeof(msg), "%ld", SendCount);
        bufferevent_write(bev, msg, strlen(msg));
        return;
    }
    bufferevent_free(bev);
}

#include "myfun.h"

/*!
    10的N次方
*/
int TMyFun::myPow10(int x) {
    int y = 1;
    if (x > 10) return 0;
    while (x-- > 0) y *= 10;
    return y;
}

/*!
    四舍五入
*/
double TMyFun::myRounded(double value, int digit) {
    double rmd;
    rmd = fmod(value *= (digit = myPow10(digit)), 1.0);
    return ((fabs(rmd) >= 0.49999 ^ rmd < 0.00) ? ceil(value) : floor(value)) / digit;
}

// #@ 3005(4位报文长度)
const int BankRateHeadLen = 6; // #@3005
//---------------------------------------------------------------------------
/**
 * @brief 基础参数(即期产品)
 */
class TSRBankBaseItem {
public:
    TSRBankBaseItem() { init(); }


    static double calcRateBySpread(double &rate, double spread, int basisPoint) {
        rate += TMyFun::myPow10(0 - basisPoint) * spread;
        return rate;
    }

    static void initArray(double *arr, int arrSize, double startValue, double step) {
        double value = startValue;
        for (int i = 0; i < arrSize; ++i) {
            value += step;
            *(arr + i) = value;
        }
    }

    void init() {
        memcpy(BankID, "003", sizeof(BankID));
        memset(CyPairName, '\0', 6);
        BasisPoint = 4;
        BaseRate = 0;
        BidFXModify = -10;
        AskFXModify = 10;

        initArray(BidModify, 9, -1, -5);
        initArray(AskModify, 9, 1, 5);
        memset(FeedCode, '\0', 6);
        memcpy(FeedCode, "   BBG", 6);

        FeedBid = 0;
        FeedAsk = 0;
        FeedMid = 0;

        initArray(BankBid, 9, 0, 0);
        initArray(BankAsk, 9, 0, 0);
        initArray(BankMid, 9, 0, 0);
    }

    virtual ~TSRBankBaseItem() = default;

    // 随机计算形成牌价
    void randomRate() {
        std::random_device sd;//生成random_device对象sd做种子
        std::minstd_rand linearRan(
                sd());//使用种子初始化linear_congruential_engine对象，为的是使用它来做我们下面随机分布的种子以输出多个随机分布.注意这里要使用()操作符，因为minst_rand()接受的是一个值（你用srand也是给出这样的一个值）
        std::uniform_int_distribution<int> dis1(0, 1000);//生成01序列
        double rate = BaseRate;
        int spread = dis1(linearRan);
        calcRateBySpread(rate, spread, BasisPoint);

        FeedBid = rate;
        FeedAsk = rate;
        calcRateBySpread(FeedBid, BidFXModify, BasisPoint);
        calcRateBySpread(FeedAsk, AskFXModify, BasisPoint);
        for (int i = 0; i < 9; ++i) {
            BankBid[i] = rate;
            calcRateBySpread(BankBid[i], BidModify[i], BasisPoint);
            calcRateBySpread(BankAsk[i], AskModify[i], BasisPoint);
            BankMid[i] = TMyFun::myRounded((BankBid[i] + BankAsk[i]) / 2, 5);
        }
    }


    /**
     * 更新data缓存
     * @param data  出参
     * @param pos 更新起点位置
     * @param content 内容
     * @param len 长度
     */
    static void addData(char *data, size_t &pos, const char *content, size_t len) {
        memcpy(data + pos, content, len);
        pos += len;
    }

    /**
     * 更新对象列表
     * @param data 数据字符串
     * @param pos data起始位置
     * @param out_lst 出参,对象列表
     */
    static long load(char *data, size_t &pos, std::unordered_map<std::string, TSRBankBaseItem *> &rateList) {
        const size_t bankIDLength = 3;
        const size_t cyPairLength = 6;
        size_t count = 0;
        // 8位牌价序号
        long rateNo = TMyFun::parseLong(data + pos, 8);
        pos += 8;

        // count: 3位长度;}
        count = TMyFun::parseInt(data + pos, 3);
        pos += 3;
        TSRBankBaseItem *pItem = nullptr;
        for (size_t i = 0; i < count; ++i) {
            char bankid[bankIDLength + 1] = {'\0'};
            memcpy(bankid, data + pos, bankIDLength);
            pos += bankIDLength;

            char cyPair[7] = {'\0'};
            memcpy(cyPair, data + pos, cyPairLength);
            pos += cyPairLength;

            auto iter = rateList.find(cyPair);
            if (iter == rateList.end()) {
                std::cout << "ratelist null" << std::endl;
                pItem = new TSRBankBaseItem();
                rateList.emplace(std::string(cyPair), pItem);
                memcpy(pItem->BankID, bankid, bankIDLength);
                memcpy(pItem->CyPairName, cyPair, cyPairLength);
            } else{
                pItem = &(*iter->second);
            }

            pItem->BasisPoint = TMyFun::parseInt(data + pos, 1);
            pos += 1;

            memcpy(pItem->FeedCode, data + pos, sizeof(pItem->FeedCode));
            pos += 6;

            pItem->FeedBid = TMyFun::parseDouble(data + pos, 16, 5);
            pos += 16;
            pItem->FeedAsk = TMyFun::parseDouble(data + pos, 16, 5);
            pos += 16;
            // 更新牌价
            pItem->updateRate();
        }
        return rateNo;
    }

    // 更新牌价
    void updateRate() {
        for (int i = 0; i < 9; ++i) {

            BankBid[i] = FeedBid;
            calcRateBySpread(BankBid[i], BidModify[i], BasisPoint);

            BankAsk[i] = FeedAsk;
            calcRateBySpread(BankAsk[i], AskModify[i], BasisPoint);
            BankMid[i] = TMyFun::myRounded((BankBid[i] + BankAsk[i]) / 2, 5);
        }
    }

    /**
     * 打包市场牌价发布
     * @param data 出参
     * @param lst 变化的行情列表
     */
    static void packRate(char *total_data, long rateNo,
                         std::unordered_map<std::string, TSRBankBaseItem *> rateList) {
        /**
         *   打包成字符串,格式: head + data
         *   1.head: @# + 4位报文长度(不含head)
         *   2.data: bankid,cypairname,BasisPoint,FeedCode,FeedBid,FeedAsk;
         */
        char buff[17];
        char *data = total_data + BankRateHeadLen;
        size_t pos = 0;
        // 00000001 (8位序号)
        memset(buff, 0, sizeof(buff));
        sprintf(buff, "%08ld", rateNo);
        addData(data, pos, buff, strlen(buff));
        // 牌价个数 (3位个数)
        memset(buff, 0, sizeof(buff));
        sprintf(buff, "%03zu", rateList.size());
        addData(data, pos, buff, strlen(buff));
        // 3,6,1,6,16.8
        // data区域:bankid,cypaiitem,BasisPoint,FeedCode,FeedBid,FeedAsk;
        for (auto iter = rateList.begin(); iter != rateList.end(); iter++) {
            TSRBankBaseItem *pItem = &(*(iter->second));
            addData(data, pos, pItem->BankID, sizeof(pItem->BankID));
            addData(data, pos, pItem->CyPairName, sizeof(pItem->CyPairName));

            memset(buff, '\0', sizeof(buff));
            sprintf(buff, "%d", pItem->BasisPoint);
            addData(data, pos, buff, strlen(buff));

            addData(data, pos, pItem->FeedCode, sizeof(pItem->FeedCode));

            memset(buff, 0, sizeof(buff));
            sprintf(buff, "%16.8f", pItem->FeedBid);
            addData(data, pos, buff, strlen(buff));

            memset(buff, 0, sizeof(buff));
            sprintf(buff, "%16.8f", pItem->FeedAsk);
            addData(data, pos, buff, strlen(buff));
        }
        // 组head
        size_t len = strlen(data);
        char tmp[BankRateHeadLen + 1];
        memset(tmp, 0, sizeof(tmp));
        sprintf(tmp, "#%05zu", len);
        memcpy(total_data, tmp, BankRateHeadLen);
    }

    /**
    * 打包银行牌价发布
    * @param data 出参
    * @param lst 变化的行情列表
    * @return 发布包的长度
    */
    static void packBankPubRate(char *total_data, long rateNo,
                                std::unordered_map<std::string, TSRBankBaseItem *> rateList) {
        // 预留出头的长度
        char *data = total_data + BankRateHeadLen;
        size_t pos = 0;
        int total = rateList.size();

        fmt::format_to_n(data+pos, 8, "{:08d}", rateNo);
        pos += 8;

        fmt::format_to_n(data+pos, 3, "{:03d}", total);
        pos += 3;
        for (std::unordered_map<std::string, TSRBankBaseItem *>::const_iterator
                     iter = rateList.begin(); iter != rateList.end(); iter++) {
            fmt::format_to_n(data+pos, 3, "{}", iter->second->BankID);
            pos += 3;
            fmt::format_to_n(data+pos, 6, "{}", iter->second->CyPairName);
            pos += 6;
            fmt::format_to_n(data+pos, 1, "{}", iter->second->BasisPoint);
            pos += 1;
            for (int i = 0; i < 9; ++i) {
                fmt::format_to_n(data+pos, 16, "{:16.8f}", iter->second->BankBid[i]);
                pos += 16;
                fmt::format_to_n(data+pos, 16, "{:16.8f}", iter->second->BankAsk[i]);
                pos += 16;
                fmt::format_to_n(data+pos, 16, "{:16.8f}", iter->second->BankMid[i]);
                pos += 16;
            }
        }
//        dataLen += rateList.size() * 442;
        char tmp[BankRateHeadLen + 1] = {'0'};
        sprintf(tmp, "#%05zu", strlen(data));
        memcpy(total_data, tmp, BankRateHeadLen);
    }

    char BankID[3];          ///<授权报价行
    char CyPairName[6];      ///<货币对名称
    int BasisPoint;         // 基点位数
    double BaseRate;        // 基准价

    char FeedCode[6];      ///<价源代码
    double FeedBid;        ///<核心买价（点差前，未处理FeedAsk价差规则)
    double FeedAsk;        ///<核心卖价（点差前未处理价差规则)
    double FeedMid;

    double BidFXModify;  // 买价固定点差
    double AskFXModify;  // 卖价固定点差
    double BidModify[9];    ///<买价点差
    double AskModify[9];    ///<卖价点差
    double BankBid[9];  ///<银行买价
    double BankAsk[9];  ///<银行卖价
    double BankMid[9];  ///<银行中间价

};

# IntervalTimer.h

#include <iostream>
#include <chrono>
#include <string>
#include <thread>

class IntervalTimer
{
public:
    typedef enum {
        ttS = 's',  // 秒
        ttMS = 'm', // 毫秒
        ttUS = 'u'  // 微秒
    }TimeType;
private:
    std::chrono::steady_clock::time_point t1;
    std::chrono::steady_clock::time_point t2;
    std::string mark;
    TimeType timeType;
public:
    IntervalTimer(std::string mark, TimeType tt = ttMS) {
        this->mark = mark;
        this->timeType = tt;
        t1 = std::chrono::steady_clock::now();
    }
    ~IntervalTimer()
    {
        t2 = std::chrono::steady_clock::now();
        if (timeType == ttS){
            auto dTimeSpan = std::chrono::duration<double, std::ratio<1>>(t2 - t1);
            std::cout << "[" << mark << "]time span: " << dTimeSpan.count() << " s\n";
        } else if (timeType == ttMS){
            auto dTimeSpan = std::chrono::duration<double, std::milli>(t2 - t1);
            std::cout << "[" << mark << "]time span: " << dTimeSpan.count() << " ms\n";
        }else if (timeType == ttUS){
            auto dTimeSpan = std::chrono::duration<double, std::micro>(t2 - t1);
            std::cout << "[" << mark << "]time span: " << dTimeSpan.count() << " us\n";
        }

    }
};
