# 数据库连接指定

 | option | memo                                                     |
 |--------|----------------------------------------------------------|
 | d      | database, 部分场景直接写数据库名，不需要d选项,但需在最前 |
 | U      | username                                                 |
 | W      | password                                                 |
 | h      | host,ip                                                  |
 | p      | port                                                     |
 | n      | schema                                                   |

# 数据库交互方式

| option | memo                 |
|--------|----------------------|
| a      | only data            |
| v      | verbose              |
| F      | format - only t: tar |

#!/bin/bash
#####################
#author : cy
#GuassDB数据库导入导出工具
#使用说明: (导入时,自动禁止触发器,导入完成后会恢复触发器)
#
#####################
#版本号
VERSION="2025-01-22-02"

# 定义帮助信息函数
show_help() {
  echo -e "Usage: $0 [OPTIONS]:"
  echo -e "\nGeneral database options:"
  echo -e "-d, --database=DATABASE                     DATABASE name"
  echo -e "-h, --host=HOSTNAME                         database server host or socket directory"
  echo -e "-p, --port=PORT                             database server port number"
  echo -e "-U, --username=NAME                         connect as specified database user"
  echo -e "-W, --password=PASSWORD                     the password of specified database user"
  #echo -e "-n, --schema=SCHEMA                         the schema to import to"
  #echo -e "-D, --datapath=DATAPATH                     datapath for saving data file"

  echo -e "\nGeneral exp/imp options:"
  echo -e "-m, --mode=EXP/IMP                          output file or directory name"
  echo -e "-f, --file=FILENAME                         exp mode:table list file\n imp mode:package name"
  #echo -e "-a, --data-only                         	   export data only, only useful in export mode"

  echo -e "\nOther options:"
  echo -e "-H, --help                         		show help"
  echo -e "\n\n************* exampls ***************"
  echo -e "for help: \n$0 -H"
  echo -e "1.exp example: \n$0 -d qes_cs -U qe_pe -W qe_pe -h 192.168.193.77 -p 30100 -m EXP -f qe_p.lst"
  echo -e "2.imp example: \n$0 -d qes -U qe_p -W qe_p -h 192.168.193.128 -p 30100 -m IMP -f qe_p_EXP_20250122_2100.tar.gz"
  exit
}
#继续or退出函数
continue_or_quit()
{
	echo -e "请确认是否继续 Y/y-继续, Q/q-退出"|tee -a $LOG
    read x
	case $x in
		"Y"|"y")
			echo -e "用户选择继续"|tee -a $LOG
			;;
		"Q"|"q")
			echo -e "用户选择退出"|tee -a $LOG
			exit
			;;
        *)
			echo -e "请重新选择 ..."|tee -a $LOG
			continue_or_quit
			;;
	esac
}

Truncate_Table()
{
	echo -e "--generate truncate sql Generated at $DT" > $SQL_truncate
	loop=0
	for table in `cat $LST|egrep -v "^$|#"`
	do
		echo -e "truncate table $table;" >> $SQL_truncate
		echo -e "alter table $table disable trigger all;\n" >> $SQL_truncate
		(( loop = loop + 1 ))
	done
	echo -e "--finish generate truncate sql Generation,total|$loop|" >> $SQL_truncate
	cat $SQL_truncate
	echo -e "Warning: finish generating truncate sql,确认清空以上表的数据?..." 
	continue_or_quit
    gsql -d $database -U $username -W $password -h $host -p $port <<-EOF
	\i ${SQL_truncate}
	EOF
    if [ $? -ne 0 ];then
        echo "Truncate_Table fail";exit
    fi
}
#恢复表的触发器
Restore_Table()
{
	echo -e "--generate restore sql Generated at $DT" > $SQL_restore
	loop=0
	for table in `cat $LST|egrep -v "^$|#"`
	do
		echo -e "alter table $table enable trigger all;\n" >> $SQL_restore
		(( loop = loop + 1 ))
	done
	
	echo -e "--finish generate restore sql Generation,total|$loop|" >> $SQL_restore
	cat $SQL_restore
    gsql -d $database -U $username -W $password -h $host -p $port <<-EOF
	\i ${SQL_restore}
	EOF

#		read x
#	fi

#	if [ $err_count -ne 0 ];then
#		clear

#		read x
#	fi
}

#去除
TransListFile()
{
	echo -e "modify ListFile ..."
	tr -d '\r' < $LST > $LST.tmp
	mv ${LST}.tmp ${LST}	
}

do_exp() {
	#建立EXP时的SHELL工作目录
	TMPDIR=${FILE_MARK}
	mkdir -p ${TMPDIR}
	if [ ! -d ${TMPDIR} ];then
		echo -e "ERROR: Create TMPDIR |$TMPDIR| FAIL ..." 
		exit
	fi
	echo -e "transListFile ..."
	TransListFile
	cp $LST ${TMPDIR}
	
	cd ${TMPDIR}
    if [ $? -ne 0 ];then
        echo "in exp, cd to ${TMPDIR} fail";exit
    fi
    EXP_DIR=`pwd`
    #read x
	echo -e "VERSIOIN|$VERSION|,now time|$DT|,args|$1,$2|..." > $LOG
	echo -e "cd to tmppath:|`pwd`|" |tee -a $LOG
    clear
    total=`cat $LST|grep -v "^$"|wc -l |awk '{print $1}'`
    loop=1
	for table in `cat $LST|egrep -v "^$|#"`
	do
		echo -e "\n[$loop/$total] exp table $table ..."
		# with tar
		#gs_dump $database -U $username -W $password -h $host -p $port --format=t -t $table -f $table.tar
        # gsql -h 192.168.193.77 -p 30100 -U qeecas -W qeecas -d qes_cs -c "\copy ecas_appl TO stdout WITH CSV HEADER;" > /home/mdata/dmp/copy/ecas_appl
        gsql -h $host -p $port -U $username -W $password -d $database -c "\copy $table TO stdout WITH CSV HEADER;" > ${EXP_DIR}/${table}.csv
        if [ $? -ne 0 ];then
            echo -e "ERROR: exp fail, table:$table" |tee -a $LOG
            exit
        fi
		(( loop = loop + 1 ))
	done
	echo -e "==================== tar ..." |tee -a $LOG
	echo -e "LOG|$LOG|"
	tar zcvf $TMPDIR.tar.gz * |xargs ls -ltr |tee -a $LOG
	mv $TMPDIR.tar.gz ../
	cd ../
	rm -rf $TMPDIR
}
do_imp() {
	########### IMP 模式下需先清空表,禁止触发器,导入完成后恢复
	SQL_truncate=trun_${FILE_MARK}.sql 
	SQL_restore=rest_${FILE_MARK}.sql

	#若是导入则新建导入临时目录
	IMP_TMPDIR=imp_${FILE_MARK}

	#如果是IMP，建立临时目录，解压、解tar
	mkdir $IMP_TMPDIR
	echo -e "导入模式新建临时目录:${IMP_TMPDIR}" 
	cp -p $file $IMP_TMPDIR

	cd $IMP_TMPDIR
    if [ $? -ne 0 ];then
        echo -e "ERROR: cd to imp tmpdir fail:$IMP_TMPDIR";exit
    fi
    IMP_DIR=`pwd`
	echo -e "VERSIOIN|$VERSION|,now time|$DT| ..." | tee -a $LOG
	echo -e "cd to tmppath:[$IMP_DIR]" |tee -a $LOG
    echo "unpack $file ..."|tee -a $LOG
	tar zxvf $file
	if [ $? -eq 0 ]; then
		echo -e "==================== unpack ok ... \n" |tee -a $LOG
	else
		echo -e "!!! ERROR !!! -----unpack fail ... \n" |tee -a $LOG
        exit
	fi
	ls -ltr |tee -a $LOG
	LST=$LST_PREF.lst

	#清空表
	Truncate_Table
	echo -e "\n==================================="
	echo -e "完成数据清理,继续数据导入IMP? ..."
	continue_or_quit
	echo -e "==================================="

	#导入-恢复数据
    total=`cat $LST|grep -v "^$"|wc -l |awk '{print $1}'`
    loop=1
	for table in `cat $LST|egrep -v "^$|#"`
	do
		echo -e "\n[$loop/$total] imp table $table ..."
		# with tar
        # gs_restore -v -d qes -U qe_e -W qe_e -h 192.168.193.128 -p 30100 -n qeecas -t ecas_white_namelist --format=t ecas_white_namelist.tar
        # gs_restore -v -d $database -U $username -W $password -h $host -p $port -n $schema --format=t -t $table $table.tar
        # gsql -h 192.168.193.128 -p 30100 -U qe_e -W qe_e -d qes -c "\copy ecas_appl FROM stdin WITH CSV HEADER;" < /home/mdata/dmp/copy/ecas_appl.csv
        gsql -h $host -p $port -U $username -W $password -d $database -c "\copy $table FROM stdin WITH CSV HEADER;" < ${IMP_DIR}/${table}.csv
        if [ $? -ne 0 ];then
            echo -e "ERROR: imp fail, table:$table" |tee -a $LOG
        fi
		(( loop = loop + 1 ))
	done

	#恢复表
    Restore_Table
	rm -fr $IMP_TMPDIR
}

### main 生成parfile
#获取SHELL当前路径
SHELL_DIR=`pwd`
#日期时间
DT=`date '+%Y%m%d_%H%M'`
# 解析命令行选项
OPTS=$(getopt -o d:h:p:U:W:n:m:f:aH --long database:,host:,port:,username:,password:,schema:,mode:,file:,data-only,help -- "$@")
OPTS=$(getopt -o d:h:p:U:W:m:f:H --long database:,host:,port:,username:,password:,mode:,file:,help -- "$@")
if [ $? -ne 0 ]; then
  echo "getopt parse options fail" >&2
  show_help
  echo "fail,OPTS: $OPTS"
  exit 1
fi
echo "parse ok,OPTS: $OPTS"

# 注意将命令行参数重置为 getopt 的输出
eval set -- "$OPTS"

# 现在处理选项
while true
do
  #echo "in parse ...,arg: $1"
  case "$1" in
	# General database options
    -d|--database)
      database="$2"
      echo "database: $database"
      shift 2
      ;;
    -h|--host)
      host="$2"
      echo "host: $host"
      shift 2
      ;;
    -p|--port)
      port="$2"
      echo "port: $port"
      shift 2
      ;;
    -U|--username)
      username="$2"
      echo "port: $port"
      shift 2
      ;;
    -W|--password)
      password="$2"
      echo "password: $password"
      shift 2
      ;;
#    -D|--datapath)
#      datapath="$2"
#      echo "datapath: $datapath"
#      shift 2
#      ;;
#    -n|--schema)
#      schema="$2"
#      echo "schema: $schema"
#      shift 2
#      ;;
	# General exp/imp options
	-m|--mode)
      mode="$2"
      echo "mode : $mode"
      shift 2
      ;;
	-f|--file)
      file="$2"
      echo "file : $file"
      shift 2
      ;;
#	-a|--data-only)
#	  dataonly = true
#	  echo "dataonly : $dataonly"
#      shift
#      ;;

    # Other options
    -h|--help)
      show_help
      shift
      exit 0
      ;;
    --)
      shift
      break
      ;;
    *)
      echo "parse internal error:$2 ." >&2
      exit 1
      ;;
  esac
done

# 处理剩余的非选项参数
echo "remain parameters:  $@"

appname=`echo -e $0|awk -F'/' '{print $NF}'`
echo -e "appname|$appname| ... "
#判断模式：导出or导入
case $mode in
	'EXP'):
        LST=${file}
		#table列表文件
		if [ ! -f ${LST} ];then
			echo -e "\n=========!!! WARNING !!! 表清单文件|${LST}| 不存在;"; exit
		fi
		echo -e "模式|数据导出 $mode|,OPTS|$OPTS|,列表清单文件|$LST|"
        # 去除末尾的换行
        echo -e "transListFile ..."
        TransListFile
        LST_PREF=`echo -e $LST|awk -F"." '{print $1}'`
        echo LST_PREF:$LST_PREF
        FILE_MARK=${LST_PREF}_${mode}_${DT}
        LOG=${appname}_${FILE_MARK}.log
        echo -e "file mark:$FILE_MARK, LST_PREF|$LST_PREF|"
        do_exp
		;;
	'IMP'):
		echo -e "============================================="
		echo -e "模式|Warning: 数据导入 $mode|,DBCONN|$OPTS|\n"
		echo -e "============================================="
        FILE_MARK=`echo -e ${file}|awk -F"." '{print $1}'`
        LST_PREF=`echo -e $FILE_MARK| awk -F"_EXP" '{print $1}'`
        LOG=${appname}_${FILE_MARK}.log
        echo -e "file mark:$FILE_MARK, LST_PREF|$LST_PREF|"
        do_imp
		;;
	*):
        show_help
		;;
esac
