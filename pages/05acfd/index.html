自定义指令directive

# 自定义指令-简单示例

<script>
    const app = Vue.createApp({
        template: `
    <div> 
    1: <input/> <br/>

    <!-- 被自动focus -->
    <input v-focus/> <br/>
    
    2: <input/> <br/>
        </div>
    `
    });
    // 全局指令 (也可以调整为局部指令,方式同mixin)
    app.directive('focus', {
        mounted(el) {
            el.focus();
        }
    });
    app.mount('#root');
</script>

# 自定义指令-动态数据传值

# 单指令

<head>
<style>
    .header { position: absolute; }
</style>
<head/>

<script>
    const app = Vue.createApp({
        data() {
            return {
                top: 120,
            }
        },
        template: `
    <div> 
    <input v-pos="top" class="header"/> <br/>
        </div>
    `
    });
    // app.directive('pos', {
    //     mounted(el, binding) {
    //         el.style.top = (binding.value + 'px');
    //     },
    //     updated(el, binding) {
    //         el.style.top = (binding.value + 'px');
    //     }
    // });
    // 以上等价于下面一个语句
    app.directive('pos', (el, binding) => {
        el.style.top = (binding.value + 'px');
    });
    app.mount('#root');
</script>

# 自定义指令加入arg

<script>
    const app = Vue.createApp({
        data() {
            return {
                distance: 220,
            }
        },
        template: `
    <div> 
    <input v-pos:left="distance" class="header"/> <br/>
        </div>
    `
    });
    app.directive('pos', (el, binding) => {
        el.style[binding.arg] = (binding.value + 'px');
    });
    app.mount('#root');
</script>
